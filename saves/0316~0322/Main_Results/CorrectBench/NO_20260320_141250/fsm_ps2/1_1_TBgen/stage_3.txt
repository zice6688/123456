def fsm_rules(input_bytes):
    """
    This function simulates the expected behavior of the FSM based on the input byte stream.
    It returns the expected 'done' signal for each clock cycle.

    :param input_bytes: A list of 8-bit bytes (integers) representing the input byte stream.
    :return: A list of boolean values representing the 'done' signal for each clock cycle.
    """
    state = 0  # Initial state
    done_signal = []  # List to store the 'done' signal for each clock cycle
    message_count = 0  # Counter for the number of bytes in the current message

    for byte in input_bytes:
        if state == 0:  # Idle state, waiting for in[3] = 1
            if (byte & 0b10000) == 0b10000:  # Check if in[3] = 1
                state = 1  # Move to state 1
                message_count = 1  # First byte of the message
            else:
                message_count = 0  # Reset message count
        elif state == 1:  # State 1, waiting for the second byte
            message_count += 1
            if message_count == 2:  # Second byte received
                state = 2  # Move to state 2
            else:
                state = 0  # Reset to idle if not the second byte
        elif state == 2:  # State 2, waiting for the third byte
            message_count += 1
            if message_count == 3:  # Third byte received
                state = 0  # Move back to idle
                done_signal.append(True)  # Signal 'done' in the next cycle
            else:
                state = 0  # Reset to idle if not the third byte
        else:
            raise ValueError("Invalid state")

        # Append the 'done' signal for the current cycle
        if len(done_signal) > 0 and done_signal[-1]:
            done_signal.append(False)  # Clear the 'done' signal after one cycle
        else:
            done_signal.append(False)

    return done_signal
