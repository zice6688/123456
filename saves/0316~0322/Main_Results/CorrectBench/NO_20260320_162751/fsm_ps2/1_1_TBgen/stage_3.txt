import numpy as np

def is_start_of_message(byte):
    """Check if the byte is the start of a message (in[3] == 1)."""
    return (byte & 0b10000) != 0

def generate_expected_done_signal(input_bytes, reset_signals):
    """Generate the expected 'done' signal based on the input bytes and reset signals.
    
    Args:
        input_bytes (list of int): List of input bytes.
        reset_signals (list of bool): List of reset signals (True for reset, False for no reset).
    
    Returns:
        list of bool: List of expected 'done' signals.
    """
    state = 'SEARCH'
    count = 0
    done_signal = []
    
    for i, (byte, reset) in enumerate(zip(input_bytes, reset_signals)):
        if reset:
            state = 'SEARCH'
            count = 0
            done_signal.append(False)
            continue
        
        if state == 'SEARCH':
            if is_start_of_message(byte):
                state = 'COUNTING'
                count = 1
            else:
                count = 0
        elif state == 'COUNTING':
            count += 1
            if count == 3:
                state = 'DONE'
        
        if state == 'DONE':
            done_signal.append(True)
            state = 'SEARCH'
            count = 0
        else:
            done_signal.append(False)
    
    return done_signal

# Example usage
input_bytes = [0b00000000, 0b00000000, 0b10000000, 0b00000000, 0b00000000]
reset_signals = [True, False, False, False, False]

expected_done_signal = generate_expected_done_signal(input_bytes, reset_signals)
print(expected_done_signal)
