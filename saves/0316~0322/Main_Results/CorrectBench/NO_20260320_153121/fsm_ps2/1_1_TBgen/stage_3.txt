import numpy as np

def is_message_start(byte):
    """Check if the byte is the start of a message (in[3] == 1)."""
    return (byte & 0x08) != 0

def generate_expected_done_signal(input_bytes):
    """Generate the expected 'done' signal for the given input byte sequence."""
    done_signal = []
    state = 'SEARCHING'
    message_count = 0
    
    for byte in input_bytes:
        if state == 'SEARCHING':
            if is_message_start(byte):
                state = 'RECEIVING_2ND_BYTE'
                message_count += 1
        elif state == 'RECEIVING_2ND_BYTE':
            state = 'RECEIVING_3RD_BYTE'
        elif state == 'RECEIVING_3RD_BYTE':
            done_signal.append(1)
            state = 'SEARCHING'
        else:
            done_signal.append(0)
    
    # Append 0s to match the length of the input sequence
    while len(done_signal) < len(input_bytes):
        done_signal.append(0)
    
    return np.array(done_signal)

# Example usage
input_sequence = [0x80, 0x00, 0x00, 0x80, 0x00, 0x00]
expected_done_signal = generate_expected_done_signal(input_sequence)
print(expected_done_signal)
