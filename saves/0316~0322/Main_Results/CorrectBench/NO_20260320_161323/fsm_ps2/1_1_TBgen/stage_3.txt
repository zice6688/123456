import numpy as np

def is_message_start(byte):
    """
    Check if the given byte is the start of a message.
    The first byte of each three-byte message always has in[3]=1.
    """
    return (byte & 0b1000) == 0b1000

def generate_expected_done_signal(input_bytes):
    """
    Generate the expected 'done' signal for the given input byte stream.
    
    :param input_bytes: List of 8-bit bytes (integers)
    :return: List of boolean values indicating the expected 'done' signal for each clock cycle
    """
    state = 'searching'
    count = 0
    done_signals = []

    for byte in input_bytes:
        if state == 'searching':
            if is_message_start(byte):
                state = 'counting'
                count = 1
            else:
                count = 0
        elif state == 'counting':
            count += 1
            if count == 3:
                done_signals.append(True)
                state = 'searching'
            else:
                done_signals.append(False)
        else:
            raise ValueError("Invalid state")

    # Pad the done_signals list to match the length of input_bytes
    done_signals.extend([False] * (len(input_bytes) - len(done_signals)))

    return done_signals
