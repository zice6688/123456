The FSM should signal done in the cycle immediately after the third byte of each message was successfully received. Reset should be active high synchronous.

def fsm_rules(in_bytes, reset):
    """
    This function simulates the FSM rules for the PS/2 mouse protocol message boundary detection.
    
    Parameters:
    in_bytes (list of int): A list of 8-bit input bytes.
    reset (list of bool): A list of boolean values indicating the reset signal for each clock cycle.
    
    Returns:
    done (list of bool): A list of boolean values indicating the `done` signal for each clock cycle.
    """
    state = 'IDLE'
    count = 0
    done = [False] * len(in_bytes)
    
    for i, (byte, rst) in enumerate(zip(in_bytes, reset)):
        if rst:
            state = 'IDLE'
            count = 0
            continue
        
        if state == 'IDLE':
            if (byte & 0b10000) == 0b10000:  # Check if in[3] is 1
                state = 'BYTE1'
                count = 1
        elif state == 'BYTE1':
            if count == 1:
                state = 'BYTE2'
                count += 1
            elif count == 2:
                state = 'BYTE3'
                count += 1
        elif state == 'BYTE2':
            if count == 2:
                state = 'BYTE3'
                count += 1
        elif state == 'BYTE3':
            if count == 3:
                done[i] = True
                state = 'IDLE'
                count = 0
    
    return done

        in_byte = signal_vector['in']
        reset = signal_vector['reset']

        if reset:
            self.state_reg = 'IDLE'
            self.count_reg = 0
            self.done_reg = False

                self.state_reg = 'BYTE2'
                self.count_reg += 1
