import numpy as np

def is_start_of_message(byte):
    """Check if the byte is the start of a message (in[3] == 1)."""
    return (byte & 0b00001000) != 0

def generate_expected_done_signal(byte_stream):
    """Generate the expected 'done' signal for the given byte stream."""
    done_signal = []
    state = 'waiting_for_start'
    message_count = 0

    for byte in byte_stream:
        if state == 'waiting_for_start':
            if is_start_of_message(byte):
                state = 'waiting_for_second_byte'
                message_count = 1
                done_signal.append(0)
            else:
                done_signal.append(0)
        elif state == 'waiting_for_second_byte':
            if message_count == 1:
                state = 'waiting_for_third_byte'
                message_count += 1
                done_signal.append(0)
            else:
                state = 'waiting_for_start'
                done_signal.append(0)
        elif state == 'waiting_for_third_byte':
            if message_count == 2:
                state = 'message_complete'
                message_count += 1
                done_signal.append(0)
            else:
                state = 'waiting_for_start'
                done_signal.append(0)
        elif state == 'message_complete':
            done_signal.append(1)
            state = 'waiting_for_start'
            message_count = 0
        else:
            done_signal.append(0)

    return np.array(done_signal, dtype=int)

# Example usage
byte_stream = [0b00000000, 0b00001000, 0b00000000, 0b00000000, 0b00001000, 0b00000000, 0b00000000]
expected_done_signal = generate_expected_done_signal(byte_stream)
print(expected_done_signal)
