import numpy as np

def is_start_of_message(byte):
    """
    Check if the given byte is the start of a message.
    The first byte of each message always has in[3]=1.
    """
    return (byte & 0b10000) == 0b10000

def process_byte_stream(byte_stream, reset_signal):
    """
    Process the byte stream and generate the expected 'done' signal based on the FSM rules.
    
    :param byte_stream: List of bytes (integers) representing the input byte stream.
    :param reset_signal: List of boolean values indicating the reset signal for each clock cycle.
    :return: List of boolean values indicating the expected 'done' signal for each clock cycle.
    """
    state = 'IDLE'
    done_signal = []
    message_count = 0

    for i, (byte, reset) in enumerate(zip(byte_stream, reset_signal)):
        if reset:
            state = 'IDLE'
            message_count = 0
            done_signal.append(False)
            continue

        if state == 'IDLE':
            if is_start_of_message(byte):
                state = 'RECEIVING_2ND_BYTE'
                message_count = 1
            else:
                state = 'IDLE'
                message_count = 0
        elif state == 'RECEIVING_2ND_BYTE':
            state = 'RECEIVING_3RD_BYTE'
            message_count += 1
        elif state == 'RECEIVING_3RD_BYTE':
            state = 'IDLE'
            message_count += 1
            done_signal.append(True)
        else:
            done_signal.append(False)

        if message_count == 3:
            done_signal.append(True)
            state = 'IDLE'
            message_count = 0
        else:
            done_signal.append(False)

    return done_signal

# Example usage
byte_stream = [0b10000000, 0b00000000, 0b00000000, 0b10000000, 0b00000000, 0b00000000]
reset_signal = [True, False, False, False, False, False]

expected_done_signal = process_byte_stream(byte_stream, reset_signal)
print(expected_done_signal)
