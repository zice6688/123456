import numpy as np

def is_start_of_message(byte):
    # Check if the 4th bit (in[3]) of the byte is 1
    return (byte & 0b00001000) == 0b00001000

def fsm_rules(input_stream, reset):
    state = 'IDLE'
    count = 0
    done = False
    expected_done = []

    for i, byte in enumerate(input_stream):
        if reset[i]:
            state = 'IDLE'
            count = 0
            done = False
        else:
            if state == 'IDLE':
                if is_start_of_message(byte):
                    state = 'COUNTING'
                    count = 1
                    done = False
            elif state == 'COUNTING':
                count += 1
                if count == 3:
                    state = 'DONE'
                    done = True
                    expected_done.append(True)
                else:
                    expected_done.append(False)
            elif state == 'DONE':
                if is_start_of_message(byte):
                    state = 'COUNTING'
                    count = 1
                    done = False
                else:
                    state = 'IDLE'
                    count = 0
                    done = False
                    expected_done.append(False)

    return expected_done

# Example usage
input_stream = [0b00000000, 0b00010000, 0b00000000, 0b00000000]  # Example input stream
reset = [False, False, False, False]  # Example reset signal
expected_done = fsm_rules(input_stream, reset)
