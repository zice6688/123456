import numpy as np

# Define the states
WALK_LEFT = 0b0001
WALK_RIGHT = 0b0010
FALLING = 0b0100
DIGGING = 0b1000
SPLATTERED = 0b10000

# Initialize the state and counters
current_state = WALK_LEFT
fall_counter = 0

def update_state(bump_left, bump_right, ground, dig):
    global current_state, fall_counter
    
    # Reset the fall counter if not falling
    if current_state != FALLING:
        fall_counter = 0
    
    # Determine the next state based on the inputs and current state
    if ground == 0:
        if current_state in [WALK_LEFT, WALK_RIGHT, DIGGING]:
            current_state = FALLING
            fall_counter += 1
        elif current_state == FALLING:
            fall_counter += 1
            if fall_counter > 20:
                current_state = SPLATTERED
    elif ground == 1:
        if current_state == FALLING:
            if fall_counter > 20:
                current_state = SPLATTERED
            else:
                current_state = WALK_LEFT if (current_state == WALK_LEFT or current_state == DIGGING) else WALK_RIGHT
        elif current_state == DIGGING:
            current_state = WALK_LEFT if (current_state == WALK_LEFT) else WALK_RIGHT
        elif dig == 1:
            current_state = DIGGING
        elif bump_left == 1 and bump_right == 1:
            current_state = WALK_RIGHT if current_state == WALK_LEFT else WALK_LEFT
        elif bump_left == 1:
            current_state = WALK_RIGHT
        elif bump_right == 1:
            current_state = WALK_LEFT
    
    return current_state

def get_outputs(current_state):
    walk_left = 1 if current_state == WALK_LEFT else 0
    walk_right = 1 if current_state == WALK_RIGHT else 0
