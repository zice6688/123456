import numpy as np

def lemming_rules(bump_left, bump_right, ground, dig, current_state, fall_counter):
    """
    This function implements the core rules for the Lemming's behavior based on the given inputs.
    
    Parameters:
    bump_left (int): 1 if the Lemming is bumped on the left, 0 otherwise.
    bump_right (int): 1 if the Lemming is bumped on the right, 0 otherwise.
    ground (int): 1 if the Lemming is on the ground, 0 otherwise.
    dig (int): 1 if the Lemming is told to dig, 0 otherwise.
    current_state (dict): A dictionary containing the current state of the Lemming.
        - 'walk_left' (int): 1 if the Lemming is walking left, 0 otherwise.
        - 'walk_right' (int): 1 if the Lemming is walking right, 0 otherwise.
        - 'aaah' (int): 1 if the Lemming is falling, 0 otherwise.
        - 'digging' (int): 1 if the Lemming is digging, 0 otherwise.
    fall_counter (int): The number of clock cycles the Lemming has been falling.

    Returns:
    dict: A dictionary containing the expected outputs for the Lemming.
        - 'walk_left' (int): 1 if the Lemming should walk left, 0 otherwise.
        - 'walk_right' (int): 1 if the Lemming should walk right, 0 otherwise.
        - 'aaah' (int): 1 if the Lemming should say "aaah!", 0 otherwise.
        - 'digging' (int): 1 if the Lemming should be digging, 0 otherwise.
    """
    
    # Initialize the expected outputs
    expected_outputs = {
        'walk_left': 0,
        'walk_right': 0,
        'aaah': 0,
        'digging': 0
    }
    
    # Check if the Lemming is in a splattered state
    if current_state['walk_left'] == 0 and current_state['walk_right'] == 0 and current_state['aaah'] == 0 and current_state['digging'] == 0:
        return expected_outputs
    
    # Priority 1: Falling
    if ground == 0:
        if fall_counter >= 20:
            # Splatter if falling for more than 20 clock cycles and hits the ground
            if current_state['aaah'] == 1:
                return expected_outputs
        else:
            # Fall and increment the fall counter
            expected_outputs['aaah'] = 1
            return expected_outputs
    else:
        # Reset the fall counter if the Lemming is not falling
        fall_counter = 0
    
    # Priority 2: Digging
    if dig == 1 and ground == 1 and current_state['aaah'] == 0:
        expected_outputs['digging'] = 1
        return expected_outputs
    elif current_state['digging'] == 1 and ground == 0:
        # If the Lemming was digging and the ground disappears, it falls
        expected_outputs['aaah'] = 1
        return expected_outputs
    
    # Priority 3: Switching directions
    if bump_left == 1 or bump_right == 1:
        if bump_left == 1:
            expected_outputs['walk_left'] = 0
            expected_outputs['walk_right'] = 1
        elif bump_right == 1:
            expected_outputs['walk_left'] = 1
            expected_outputs['walk_right'] = 0
        return expected_outputs
    
    # Default behavior: Continue in the current direction
    expected_outputs['walk_left'] = current_state['walk_left']
    expected_outputs['walk_right'] = current_state['walk_right']
    
    return expected_outputs
