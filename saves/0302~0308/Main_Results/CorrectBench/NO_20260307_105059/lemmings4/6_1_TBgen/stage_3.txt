# Define the states

def update_state(current_state, bump_left, bump_right, ground, dig):

        return WALK_LEFT, 0, 0, 0, 0
    
    # Determine the next state and outputs

            return WALK_RIGHT, 0, 0, 0
        elif not ground:
            return FALLING, 0, 1, 0
        elif dig:
            return DIGGING, 0, 0, 1
        else:
            return WALK_LEFT, 1, 0, 0
    
    elif current_state == WALK_RIGHT:
        if bump_left or bump_right:
            return WALK_LEFT, 0, 0, 0
        elif not ground:
            return FALLING, 0, 1, 0
        elif dig:
            return DIGGING, 0, 0, 1
        else:
            return WALK_RIGHT, 0, 1, 0

                    return WALK_RIGHT, 0, 1, 0
                elif prev_state == DIGGING:
                    return DIGGING, 0, 0, 1

            return FALLING, 0, 1, 0

        else:
            return DIGGING, 0, 0, 1
