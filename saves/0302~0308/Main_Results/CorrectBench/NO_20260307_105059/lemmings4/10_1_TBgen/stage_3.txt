import numpy as np

# Define the states
WALK_LEFT = 0b0001
WALK_RIGHT = 0b0010
FALLING = 0b0100
DIGGING = 0b1000
SPLATTER = 0b10000

# Define the initial state
current_state = WALK_LEFT
falling_counter = 0

def update_state(current_state, bump_left, bump_right, ground, dig, falling_counter):
    global falling_counter
    next_state = current_state
    aaah = 0
    digging = 0
    walk_left = 0
    walk_right = 0
    
    if current_state == WALK_LEFT:
        if bump_left:
            next_state = WALK_RIGHT
        elif not ground:
            next_state = FALLING
        elif dig and ground:
            next_state = DIGGING
    elif current_state == WALK_RIGHT:
        if bump_right:
            next_state = WALK_LEFT
        elif not ground:
            next_state = FALLING
        elif dig and ground:
            next_state = DIGGING
    elif current_state == FALLING:
        if ground:
            if falling_counter > 20:
                next_state = SPLATTER
            else:
                next_state = WALK_LEFT if (current_state & WALK_LEFT) else WALK_RIGHT
        else:
            falling_counter += 1
        aaah = 1
    elif current_state == DIGGING:
        if not ground:
            next_state = FALLING
            aaah = 1
        else:
            digging = 1
    elif current_state == SPLATTER:
        pass  # Stay in SPLATTER state
    
    if next_state == WALK_LEFT:
        walk_left = 1
    elif next_state == WALK_RIGHT:
        walk_right = 1
    elif next_state == DIGGING:
        digging = 1
    elif next_state == FALLING:
        aaah = 1
    elif next_state == SPLATTER:
        walk_left = 0
        walk_right = 0
        aaah = 0
        digging = 0
    
    return next_state, walk_left, walk_right, aaah, digging
