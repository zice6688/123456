import numpy as np

# Define the states
WALK_LEFT = 0b0001
WALK_RIGHT = 0b0010
FALLING = 0b0100
DIGGING = 0b1000
SPLATTER = 0b0000

# Define the outputs
def get_outputs(state):
    if state == WALK_LEFT:
        return {'walk_left': 1, 'walk_right': 0, 'aaah': 0, 'digging': 0}
    elif state == WALK_RIGHT:
        return {'walk_left': 0, 'walk_right': 1, 'aaah': 0, 'digging': 0}
    elif state == FALLING:
        return {'walk_left': 0, 'walk_right': 0, 'aaah': 1, 'digging': 0}
    elif state == DIGGING:
        return {'walk_left': 0, 'walk_right': 0, 'aaah': 0, 'digging': 1}
    elif state == SPLATTER:
        return {'walk_left': 0, 'walk_right': 0, 'aaah': 0, 'digging': 0}

# Core rules for the DUT
def lemming_rules(current_state, bump_left, bump_right, ground, dig, fall_counter):
    next_state = current_state
    new_fall_counter = fall_counter

    if current_state == WALK_LEFT:
        if bump_left:
            next_state = WALK_RIGHT
        elif bump_right:
            next_state = WALK_RIGHT
        elif not ground:
            next_state = FALLING
            new_fall_counter = 0
        elif dig and ground:
            next_state = DIGGING
    elif current_state == WALK_RIGHT:
        if bump_left:
            next_state = WALK_LEFT
        elif bump_right:
            next_state = WALK_LEFT
        elif not ground:
            next_state = FALLING
            new_fall_counter = 0
        elif dig and ground:
            next_state = DIGGING
    elif current_state == FALLING:
        if ground:
            if fall_counter < 20:
                if current_state == WALK_LEFT:
                    next_state = WALK_LEFT
                else:
                    next_state = WALK_RIGHT
            else:
                next_state = SPLATTER
            new_fall_counter = 0
        else:
            new_fall_counter += 1
    elif current_state == DIGGING:
        if not ground:
            next_state = FALLING
            new_fall_counter = 0
        elif ground:
            if current_state == WALK_LEFT:
                next_state = WALK_LEFT
            else:
                next_state = WALK_RIGHT
    elif current_state == SPLATTER:
        pass  # No change in state

    return next_state, new_fall_counter
