def next_state_and_outputs(current_state, bump_left, bump_right, ground, dig):
    next_state = current_state
    walk_left = 0
    walk_right = 0
    aaah = 0
    digging = 0
    
    if current_state == WL:
        if bump_left:
            next_state = WR
        elif not ground:
            next_state = F
            aaah = 1
        elif dig:
            next_state = D
            digging = 1
        else:
            walk_left = 1
    elif current_state == WR:
        if bump_right:
            next_state = WL
        elif not ground:
            next_state = F
            aaah = 1
        elif dig:
            next_state = D
            digging = 1
        else:
            walk_right = 1
    elif current_state == F:
        if ground:
            if current_state == F:
                next_state = S
            else:
                next_state = current_state - F + (current_state & (WL | WR))
            aaah = 0
        else:
            aaah = 1

            next_state = F
            aaah = 1
        else:
            digging = 1
    elif current_state == S:
        pass  # Stay in the splattered state
    
    return next_state, walk_left, walk_right, aaah, digging
