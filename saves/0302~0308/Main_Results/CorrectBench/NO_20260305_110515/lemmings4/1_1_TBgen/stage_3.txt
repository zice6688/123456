import numpy as np

def lemming_rules(bump_left, bump_right, ground, dig, current_state, fall_counter):
    # Initialize outputs
    walk_left = 0
    walk_right = 0
    aaah = 0
    digging = 0
    
    # Determine the current state
    if current_state == 'walk_left':
        if bump_left and not bump_right:
            next_state = 'walk_right'
        elif bump_right and not bump_left:
            next_state = 'walk_left'
        elif bump_left and bump_right:
            next_state = 'walk_right'
        else:
            next_state = 'walk_left'
    elif current_state == 'walk_right':
        if bump_left and not bump_right:
            next_state = 'walk_right'
        elif bump_right and not bump_left:
            next_state = 'walk_left'
        elif bump_left and bump_right:
            next_state = 'walk_left'
        else:
            next_state = 'walk_right'
    elif current_state == 'falling':
        if ground:
            if fall_counter > 20:
                next_state = 'splattered'
            else:
                next_state = current_state
        else:
            next_state = 'falling'
    elif current_state == 'digging':
        if not ground:
            next_state = 'falling'
        else:
            next_state = 'digging'
    elif current_state == 'splattered':
        next_state = 'splattered'
    
    # Update outputs based on the next state
    if next_state == 'walk_left':
        walk_left = 1
    elif next_state == 'walk_right':
        walk_right = 1
    elif next_state == 'falling':
        aaah = 1
    elif next_state == 'digging':
        digging = 1
    elif next_state == 'splattered':
        pass  # All outputs are 0
    
    # Update fall counter
    if next_state == 'falling':
        fall_counter += 1
    else:
        fall_counter = 0
    
    return walk_left, walk_right, aaah, digging, next_state, fall_counter
