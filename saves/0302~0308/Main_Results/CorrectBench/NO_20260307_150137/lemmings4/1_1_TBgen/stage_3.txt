import numpy as np

# Define the states
WALK_LEFT = 0b0001
WALK_RIGHT = 0b0010
FALLING = 0b0100
DIGGING = 0b1000
SPLATTERED = 0b10000

# Define the outputs
WALK_LEFT_OUT = 0b0001
WALK_RIGHT_OUT = 0b0010
AAAH = 0b0100
DIGGING_OUT = 0b1000

def lemming_rules(state, bump_left, bump_right, ground, dig, fall_counter):
    next_state = state
    walk_left = 0
    walk_right = 0
    aaah = 0
    digging = 0
    
    if state == WALK_LEFT:
        if bump_left:
            next_state = WALK_RIGHT
        elif bump_right:
            next_state = WALK_LEFT
        elif not ground:
            next_state = FALLING
            aaah = 1
        elif dig and ground:
            next_state = DIGGING
            digging = 1
        else:
            walk_left = 1
    
    elif state == WALK_RIGHT:
        if bump_left:
            next_state = WALK_LEFT
        elif bump_right:
            next_state = WALK_RIGHT
        elif not ground:
            next_state = FALLING
            aaah = 1
        elif dig and ground:
            next_state = DIGGING
            digging = 1
        else:
            walk_right = 1
    
    elif state == FALLING:
        if ground:
            if fall_counter > 20:
                next_state = SPLATTERED
            else:
                if state == WALK_LEFT:
                    next_state = WALK_LEFT
                    walk_left = 1
                else:
                    next_state = WALK_RIGHT
                    walk_right = 1
        else:
            aaah = 1
    
    elif state == DIGGING:
        if not ground:
            next_state = FALLING
            aaah = 1
        elif ground:
            digging = 1
        else:
            if state == WALK_LEFT:
                next_state = WALK_LEFT
                walk_left = 1
            else:
                next_state = WALK_RIGHT
                walk_right = 1
    
    elif state == SPLATTERED:
        next_state = SPLATTERED
    
    return next_state, walk_left, walk_right, aaah, digging
