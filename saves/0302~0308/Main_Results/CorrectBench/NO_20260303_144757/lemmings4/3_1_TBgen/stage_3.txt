import numpy as np

def lemming_fsm_rules(clk, areset, bump_left, bump_right, ground, dig, current_state, fall_counter):
    # Define states
    WALK_LEFT = 0b0001
    WALK_RIGHT = 0b0010
    FALLING = 0b0100
    DIGGING = 0b1000
    SPLATTERED = 0b0000

    # Initialize outputs
    walk_left = 0
    walk_right = 0
    aaah = 0
    digging = 0

    # Asynchronous reset
    if areset:
        next_state = WALK_LEFT
        fall_counter = 0
    else:
        # State transitions
        if current_state == WALK_LEFT:
            if bump_left:
                next_state = WALK_RIGHT
            elif bump_right:
                next_state = WALK_LEFT
            elif not ground:
                next_state = FALLING
            elif dig and ground:
                next_state = DIGGING
            else:
                next_state = WALK_LEFT
        elif current_state == WALK_RIGHT:
            if bump_right:
                next_state = WALK_LEFT
            elif bump_left:
                next_state = WALK_RIGHT
            elif not ground:
                next_state = FALLING
            elif dig and ground:
                next_state = DIGGING
            else:
                next_state = WALK_RIGHT
        elif current_state == FALLING:
            if ground:
                if fall_counter > 20:
                    next_state = SPLATTERED
                else:
                    next_state = WALK_LEFT if (current_state == WALK_LEFT or current_state == DIGGING) else WALK_RIGHT
            else:
                next_state = FALLING
        elif current_state == DIGGING:
            if not ground:
                next_state = FALLING
            else:
                next_state = DIGGING
        elif current_state == SPLATTERED:
            next_state = SPLATTERED

        # Update fall counter
        if current_state == FALLING:
            fall_counter += 1
        else:
            fall_counter = 0

    # Output logic
    if next_state == WALK_LEFT:
        walk_left = 1
    elif next_state == WALK_RIGHT:
        walk_right = 1
    elif next_state == FALLING:
        aaah = 1
    elif next_state == DIGGING:
        digging = 1

    return next_state, walk_left, walk_right, aaah, digging, fall_counter

# Example usage

    clk, areset, bump_left, bump_right, ground, dig, current_state, fall_counter
