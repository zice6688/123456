import numpy as np

def lemming_rules(bump_left, bump_right, ground, dig, current_state, fall_counter):
    """
    This function implements the core rules for the Lemmings' behavior based on the given inputs.
    
    Parameters:
    - bump_left: 1 if the Lemming is bumped on the left, 0 otherwise.
    - bump_right: 1 if the Lemming is bumped on the right, 0 otherwise.
    - ground: 1 if the Lemming is on the ground, 0 otherwise.
    - dig: 1 if the Lemming should start digging, 0 otherwise.
    - current_state: A dictionary containing the current state of the Lemming.
        - 'walk_left': 1 if the Lemming is walking left, 0 otherwise.
        - 'walk_right': 1 if the Lemming is walking right, 0 otherwise.
        - 'aaah': 1 if the Lemming is falling, 0 otherwise.
        - 'digging': 1 if the Lemming is digging, 0 otherwise.
    - fall_counter: The number of clock cycles the Lemming has been falling.
    
    Returns:
    - A dictionary containing the updated state of the Lemming.
    - The updated fall counter.
    """
    
    # Initialize the new state with the current state
    new_state = current_state.copy()
    
    # If the Lemming is not falling and not digging
    if not (current_state['aaah'] or current_state['digging']):
        # Switch directions if bumped
        if bump_left == 1 and bump_right == 0:
            new_state['walk_left'] = 0
            new_state['walk_right'] = 1
        elif bump_right == 1 and bump_left == 0:
            new_state['walk_left'] = 1
            new_state['walk_right'] = 0
        elif bump_left == 1 and bump_right == 1:
            if current_state['walk_left']:
                new_state['walk_left'] = 0
                new_state['walk_right'] = 1
            else:
                new_state['walk_left'] = 1
                new_state['walk_right'] = 0
    
    # Check if the Lemming should fall
    if ground == 0:
        new_state['aaah'] = 1
        new_state['walk_left'] = 0
        new_state['walk_right'] = 0
        new_state['digging'] = 0
        fall_counter += 1
    else:
        # If the Lemming was falling and now hits the ground
        if current_state['aaah']:
            if fall_counter > 20:
                # Splatter
                new_state['walk_left'] = 0
                new_state['walk_right'] = 0
                new_state['aaah'] = 0
                new_state['digging'] = 0
                fall_counter = 0
            else:
                # Resume walking in the same direction as before the fall
                new_state['aaah'] = 0
                if current_state['walk_left'] == 1:
                    new_state['walk_left'] = 1
                    new_state['walk_right'] = 0
                else:
                    new_state['walk_left'] = 0
                    new_state['walk_right'] = 1
                fall_counter = 0
        
        # Check if the Lemming should start digging
        if dig == 1 and ground == 1:
            new_state['digging'] = 1
            new_state['walk_left'] = 0
            new_state['walk_right'] = 0
        elif dig == 0 and ground == 0 and current_state['digging']:
            new_state['digging'] = 0
            new_state['aaah'] = 1
            new_state['walk_left'] = 0
            new_state['walk_right'] = 0
            fall_counter += 1
    
    return new_state, fall_counter
