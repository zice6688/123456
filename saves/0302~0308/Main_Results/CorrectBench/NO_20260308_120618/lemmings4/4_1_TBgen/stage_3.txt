# Define the states
WALK_LEFT = 0b0001
WALK_RIGHT = 0b0010
FALLING = 0b0100
DIGGING = 0b1000

# Function to determine the next state and outputs based on the current state and inputs
def determine_next_state_and_outputs(current_state, bump_left, bump_right, ground, dig, fall_counter):
    next_state = current_state
    walk_left = 0
    walk_right = 0
    aaah = 0
    digging = 0
    
    if bump_left == 1:
        if current_state == WALK_LEFT:
            next_state = WALK_RIGHT
        elif current_state == WALK_RIGHT:
            next_state = WALK_LEFT
    elif bump_right == 1:
        if current_state == WALK_LEFT:
            next_state = WALK_RIGHT
        elif current_state == WALK_RIGHT:
            next_state = WALK_LEFT
    
    if ground == 0:
        if current_state not in [SPLATTERED]:
            next_state = FALLING
            aaah = 1
            fall_counter += 1
    else:
        if current_state == FALLING:
            if fall_counter > 20:
                next_state = SPLATTERED
            else:
                next_state = current_state & (WALK_LEFT | WALK_RIGHT)
                fall_counter = 0
        elif current_state == DIGGING:

            next_state = DIGGING

        walk_right = 1

    elif next_state == DIGGING:
        digging = 1
    elif next_state == SPLATTERED:
        walk_left = 0
        walk_right = 0
        aaah = 0
        digging = 0
    
    return next_state, walk_left, walk_right, aaah, digging, fall_counter
