import numpy as np

def lemming_rules(areset, bump_left, bump_right, ground, dig, current_state, fall_counter):
    """
    This function implements the core rules for the Lemmings state machine.
    
    Parameters:
    - areset: Asynchronous reset signal (1 for reset, 0 for no reset)
    - bump_left: Bump on the left (1 for bump, 0 for no bump)
    - bump_right: Bump on the right (1 for bump, 0 for no bump)
    - ground: Ground presence (1 for ground, 0 for no ground)
    - dig: Dig command (1 for dig, 0 for no dig)
    - current_state: Current state of the Lemming (tuple: (walk_left, walk_right, aaah, digging))
    - fall_counter: Counter for the number of clock cycles the Lemming has been falling
    
    Returns:
    - next_state: Next state of the Lemming (tuple: (walk_left, walk_right, aaah, digging))
    - next_fall_counter: Updated fall counter
    """
    walk_left, walk_right, aaah, digging = current_state
    
    # Reset condition
    if areset:
        return (1, 0, 0, 0), 0
    
    # Check for splatter condition
    if not ground and fall_counter > 20:
        return (0, 0, 0, 0), 0
    
    # Update fall counter
    if aaah:
        next_fall_counter = fall_counter + 1
    else:
        next_fall_counter = 0
    
    # Handle ground and falling
    if not ground:
        if not aaah:
            next_aaah = 1
            next_digging = 0
            next_walk_left = 0
            next_walk_right = 0
        else:
            next_aaah = 1
            next_digging = 0
            next_walk_left = 0
            next_walk_right = 0
    else:
        next_aaah = 0
        next_fall_counter = 0
        
        # Handle digging
        if dig and not aaah and not digging:
            next_digging = 1
            next_walk_left = 0
            next_walk_right = 0
        elif digging:
            next_digging = 1
            next_walk_left = 0
            next_walk_right = 0
        else:
            next_digging = 0
            
            # Handle walking and bumps
            if bump_left or bump_right:
                if bump_left and not bump_right:
                    next_walk_left = 0
                    next_walk_right = 1
                elif bump_right and not bump_left:
                    next_walk_left = 1
                    next_walk_right = 0
                else:  # Both bump_left and bump_right
                    if walk_left:
                        next_walk_left = 0
                        next_walk_right = 1
                    else:
                        next_walk_left = 1
                        next_walk_right = 0
            else:
                next_walk_left = walk_left
                next_walk_right = walk_right
    
    return (next_walk_left, next_walk_right, next_aaah, next_digging), next_fall_counter
