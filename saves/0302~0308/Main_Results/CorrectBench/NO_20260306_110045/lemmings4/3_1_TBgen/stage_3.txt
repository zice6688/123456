import numpy as np

def lemming_rules(bump_left, bump_right, ground, dig, current_state, fall_counter):
    """
    This function implements the core rules for the Lemmings state machine.
    
    Parameters:
    - bump_left: 1 if the Lemming is bumped on the left, 0 otherwise.
    - bump_right: 1 if the Lemming is bumped on the right, 0 otherwise.
    - ground: 1 if the ground is present, 0 otherwise.
    - dig: 1 if the Lemming is commanded to start digging, 0 otherwise.
    - current_state: The current state of the Lemming (binary string).
    - fall_counter: Counter for the number of clock cycles the Lemming has been falling.
    
    Returns:
    - next_state: The next state of the Lemming (binary string).
    - outputs: A dictionary containing the expected values of walk_left, walk_right, aaah, and digging.
    - fall_counter: Updated fall counter.
    """
    
    # Define states
    WALK_LEFT = 0b0001
    WALK_RIGHT = 0b0010
    FALLING = 0b0100
    DIGGING = 0b1000
    SPLATTERED = 0b0000
    
    # Initialize outputs
    outputs = {
        'walk_left': 0,
        'walk_right': 0,
        'aaah': 0,
        'digging': 0
    }
    
    # Determine the next state based on the current state and inputs
    if current_state == WALK_LEFT:
        if bump_left:
            next_state = WALK_RIGHT
        elif not ground:
            next_state = FALLING
            fall_counter = 0
        elif dig and ground:
            next_state = DIGGING
        else:
            next_state = WALK_LEFT
    elif current_state == WALK_RIGHT:
        if bump_right:
            next_state = WALK_LEFT
        elif not ground:
            next_state = FALLING
            fall_counter = 0
        elif dig and ground:
            next_state = DIGGING
        else:
            next_state = WALK_RIGHT
    elif current_state == FALLING:
        if ground:
            if fall_counter >= 20:
                next_state = SPLATTERED
            else:
                if current_state == WALK_LEFT:
                    next_state = WALK_LEFT
                else:
                    next_state = WALK_RIGHT
        else:
            next_state = FALLING
            fall_counter += 1
    elif current_state == DIGGING:
        if not ground:
            next_state = FALLING
            fall_counter = 0
        else:
            next_state = DIGGING
    elif current_state == SPLATTERED:
        next_state = SPLATTERED
    else:
        next_state = SPLATTERED  # Default to splattered if in an unknown state
    
    # Set the outputs based on the next state
    if next_state == WALK_LEFT:
        outputs['walk_left'] = 1
    elif next_state == WALK_RIGHT:
        outputs['walk_right'] = 1
    elif next_state == FALLING:
        outputs['aaah'] = 1
    elif next_state == DIGGING:
        outputs['digging'] = 1
    elif next_state == SPLATTERED:
        outputs['walk_left'] = 0
        outputs['walk_right'] = 0
        outputs['aaah'] = 0
        outputs['digging'] = 0
    
    return next_state, outputs, fall_counter
