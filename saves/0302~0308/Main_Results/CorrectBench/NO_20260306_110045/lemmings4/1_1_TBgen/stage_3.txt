def lemming_rules(bump_left, bump_right, ground, dig, current_state, fall_counter):
    # Define the initial state
    if current_state is None:
        current_state = {
            'walk_left': 1,
            'walk_right': 0,
            'aaah': 0,
            'digging': 0,
            'direction': 'left',
            'falling': False,
            'splattered': False
        }
    
    # Reset the fall counter if the Lemming is not falling
    if not current_state['falling']:
        fall_counter = 0
    
    # Determine the new state based on the inputs and current state
    if bump_left == 1 or bump_right == 1:
        if current_state['direction'] == 'left':
            current_state['direction'] = 'right'
        else:
            current_state['direction'] = 'left'
        current_state['walk_left'] = 1 if current_state['direction'] == 'left' else 0
        current_state['walk_right'] = 1 if current_state['direction'] == 'right' else 0
        current_state['aaah'] = 0
        current_state['digging'] = 0
        current_state['falling'] = False
        current_state['splattered'] = False
    elif ground == 0:
        if not current_state['falling']:
            fall_counter = 0
        fall_counter += 1
        if fall_counter > 20 and (current_state['falling'] and ground == 1):
            current_state['splattered'] = True
            current_state['walk_left'] = 0
            current_state['walk_right'] = 0
            current_state['aaah'] = 0
            current_state['digging'] = 0
            current_state['falling'] = False
        else:
            current_state['walk_left'] = 0
            current_state['walk_right'] = 0
            current_state['aaah'] = 1
            current_state['digging'] = 0
            current_state['falling'] = True

            current_state['falling'] = False
