import numpy as np

def lemming_fsm_rules(bump_left, bump_right, ground, dig, current_state, fall_counter):
    # Define the states
    WALK_LEFT = 0b0001
    WALK_RIGHT = 0b0010
    FALLING = 0b0100
    DIGGING = 0b1000
    SPLATTERED = 0b10000

    # Initialize outputs
    walk_left = 0
    walk_right = 0
    aaah = 0
    digging = 0

    # Determine the next state and outputs based on the current state and inputs
    if current_state == WALK_LEFT:
        if bump_left == 1:
            next_state = WALK_RIGHT
        elif bump_right == 1:
            next_state = WALK_LEFT
        elif ground == 0:
            next_state = FALLING
            aaah = 1
        elif dig == 1:
            next_state = DIGGING
        else:
            next_state = WALK_LEFT
            walk_left = 1

    elif current_state == WALK_RIGHT:
        if bump_left == 1:
            next_state = WALK_LEFT
        elif bump_right == 1:
            next_state = WALK_RIGHT
        elif ground == 0:
            next_state = FALLING
            aaah = 1
        elif dig == 1:
            next_state = DIGGING
        else:
            next_state = WALK_RIGHT
            walk_right = 1

    elif current_state == FALLING:
        if ground == 1:
            if fall_counter > 20:
                next_state = SPLATTERED
            else:
                next_state = current_state & (WALK_LEFT | WALK_RIGHT)
                if current_state & WALK_LEFT:
                    walk_left = 1
                elif current_state & WALK_RIGHT:
                    walk_right = 1
        else:
            next_state = FALLING
            aaah = 1
            fall_counter += 1

    elif current_state == DIGGING:
        if ground == 0:
            next_state = FALLING
            aaah = 1
        else:
            next_state = DIGGING
            digging = 1

    elif current_state == SPLATTERED:
        next_state = SPLATTERED

    return next_state, walk_left, walk_right, aaah, digging, fall_counter
