import numpy as np

def lemming_fsm_rules(bump_left, bump_right, ground, dig, current_state, fall_counter):
    # Define the states
    WALK_LEFT = 0b0001
    WALK_RIGHT = 0b0010
    FALLING = 0b0100
    DIGGING = 0b1000
    SPLATTERING = 0b0000

    # Initialize the next state and outputs
    next_state = current_state
    walk_left = 0
    walk_right = 0
    aaah = 0
    digging = 0

    # Transition rules
    if current_state == WALK_LEFT:
        if bump_left:
            next_state = WALK_RIGHT
        elif bump_right:
            next_state = WALK_RIGHT
        elif not ground:
            next_state = FALLING
        elif dig and ground:
            next_state = DIGGING

    elif current_state == WALK_RIGHT:
        if bump_right:
            next_state = WALK_LEFT
        elif bump_left:
            next_state = WALK_LEFT
        elif not ground:
            next_state = FALLING
        elif dig and ground:
            next_state = DIGGING

    elif current_state == FALLING:
        if ground:
            if fall_counter > 20:
                next_state = SPLATTERING
            else:
                if current_state == FALLING and (bump_left or bump_right):
                    next_state = current_state
                else:
                    next_state = WALK_LEFT if (current_state == WALK_LEFT) else WALK_RIGHT
        else:
            fall_counter += 1

    elif current_state == DIGGING:
        if not ground:
            next_state = FALLING
        elif bump_left or bump_right:
            next_state = current_state

    elif current_state == SPLATTERING:
        pass  # Stay in splattering state

    # Output values based on the next state
    if next_state == WALK_LEFT:
        walk_left = 1
    elif next_state == WALK_RIGHT:
        walk_right = 1
    elif next_state == FALLING:
        aaah = 1
    elif next_state == DIGGING:
        digging = 1
    elif next_state == SPLATTERING:
        walk_left = 0
        walk_right = 0
        aaah = 0
        digging = 0

    return next_state, walk_left, walk_right, aaah, digging, fall_counter
