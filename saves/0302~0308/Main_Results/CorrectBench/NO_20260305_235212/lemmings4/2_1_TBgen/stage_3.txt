import numpy as np

def lemming_state_machine(bump_left, bump_right, ground, dig, current_state, fall_counter):
    """
    This function simulates the Lemming's state machine and returns the expected outputs based on the inputs.
    
    :param bump_left: 1 if the Lemming is bumped from the left, 0 otherwise
    :param bump_right: 1 if the Lemming is bumped from the right, 0 otherwise
    :param ground: 1 if there is ground, 0 otherwise
    :param dig: 1 if the Lemming should start digging, 0 otherwise
    :param current_state: The current state of the Lemming (0: walk_left, 1: walk_right, 2: falling, 3: digging, 4: splattered)
    :param fall_counter: The number of clock cycles the Lemming has been falling
    :return: A tuple (walk_left, walk_right, aaah, digging, new_state, new_fall_counter)
    """
    # Define the states
    WALK_LEFT = 0
    WALK_RIGHT = 1
    FALLING = 2
    DIGGING = 3
    SPLATTERED = 4
    
    # Initialize the outputs
    walk_left = 0
    walk_right = 0
    aaah = 0
    digging = 0
    new_state = current_state
    new_fall_counter = fall_counter
    
    # Handle reset and initial state
    if current_state == -1:
        new_state = WALK_LEFT
        walk_left = 1
        return (walk_left, walk_right, aaah, digging, new_state, new_fall_counter)
    
    # Handle falling
    if ground == 0:
        if current_state != FALLING:
            new_state = FALLING
            new_fall_counter = 1
        else:
            new_fall_counter += 1
        aaah = 1
        if new_fall_counter > 20 and ground == 1:
            new_state = SPLATTERED
            new_fall_counter = 0
            aaah = 0
            return (walk_left, walk_right, aaah, digging, new_state, new_fall_counter)
    else:
        if current_state == FALLING:
            new_state = current_state - 2  # Resume previous state
            new_fall_counter = 0
            aaah = 0
        elif current_state == SPLATTERED:
            return (walk_left, walk_right, aaah, digging, new_state, new_fall_counter)
    
    # Handle digging
    if ground == 1 and dig == 1 and current_state not in [FALLING, SPLATTERED]:
        new_state = DIGGING
        digging = 1
    elif ground == 0 and current_state == DIGGING:
        new_state = FALLING
        digging = 0
        aaah = 1
        new_fall_counter = 1
    
    # Handle walking and direction changes
    if current_state == WALK_LEFT:
        walk_left = 1
        if bump_left == 1 or (bump_left == 1 and bump_right == 1):
            new_state = WALK_RIGHT
            walk_left = 0
            walk_right = 1
    elif current_state == WALK_RIGHT:
        walk_right = 1
        if bump_right == 1 or (bump_left == 1 and bump_right == 1):
            new_state = WALK_LEFT
            walk_right = 0
            walk_left = 1
    
    return (walk_left, walk_right, aaah, digging, new_state, new_fall_counter)
