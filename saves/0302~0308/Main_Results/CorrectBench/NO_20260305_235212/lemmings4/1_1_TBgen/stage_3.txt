def lemming_rules(clk, areset, bump_left, bump_right, ground, dig, current_state, fall_counter):
    # Define the states
    WALK_LEFT = 0b0001
    WALK_RIGHT = 0b0010
    FALLING = 0b0100
    DIGGING = 0b1000
    SPLATTERED = 0b10000

    # Initialize outputs
    walk_left = 0
    walk_right = 0
    aaah = 0
    digging = 0

    # Asynchronous reset
    if areset:
        current_state = WALK_LEFT
        fall_counter = 0
        return (current_state, fall_counter, walk_left, walk_right, aaah, digging)

    # State transitions and output generation
    if current_state == WALK_LEFT:
        if bump_left:
            current_state = WALK_RIGHT
        elif bump_right:
            current_state = WALK_LEFT  # No change, but it's still a valid transition
        elif not ground:
            current_state = FALLING
            aaah = 1
        elif dig and ground:
            current_state = DIGGING
            digging = 1
        else:
            walk_left = 1

    elif current_state == WALK_RIGHT:
        if bump_right:
            current_state = WALK_LEFT
        elif bump_left:
            current_state = WALK_RIGHT  # No change, but it's still a valid transition
        elif not ground:
            current_state = FALLING
            aaah = 1
        elif dig and ground:
            current_state = DIGGING
            digging = 1
        else:
            walk_right = 1

    elif current_state == FALLING:
        if not ground:
            fall_counter += 1
            aaah = 1
        else:
            if fall_counter > 20:
                current_state = SPLATTERED
            else:
                if current_state == WALK_LEFT:
                    current_state = WALK_LEFT
                    walk_left = 1
                elif current_state == WALK_RIGHT:
                    current_state = WALK_RIGHT
                    walk_right = 1
            fall_counter = 0
            aaah = 0

    elif current_state == DIGGING:
        if not ground:
            current_state = FALLING
            aaah = 1
            digging = 0
        else:
            digging = 1

    elif current_state == SPLATTERED:
        pass  # All outputs remain 0

    return (current_state, fall_counter, walk_left, walk_right, aaah, digging)
