import numpy as np

def lemming_rules(bump_left, bump_right, ground, dig, current_state, fall_counter):
    """
    This function implements the core rules for the Lemming's behavior based on the given inputs.
    
    Parameters:
    - bump_left: 1 if the Lemming is bumped on the left, 0 otherwise.
    - bump_right: 1 if the Lemming is bumped on the right, 0 otherwise.
    - ground: 1 if the ground is present, 0 otherwise.
    - dig: 1 if the Lemming should start digging, 0 otherwise.
    - current_state: A dictionary containing the current state of the Lemming.
        - 'walk_left': 1 if the Lemming is walking left, 0 otherwise.
        - 'walk_right': 1 if the Lemming is walking right, 0 otherwise.
        - 'aaah': 1 if the Lemming is falling and saying "aaah!", 0 otherwise.
        - 'digging': 1 if the Lemming is digging, 0 otherwise.
    - fall_counter: The number of clock cycles the Lemming has been falling.
    
    Returns:
    - A dictionary containing the expected values for the Lemming's outputs.
    """
    
    # Initialize the expected output dictionary
    expected_outputs = {
        'walk_left': 0,
        'walk_right': 0,
        'aaah': 0,
        'digging': 0
    }
    
    # Determine the current walking direction
    walking_left = current_state['walk_left']
    walking_right = current_state['walk_right']
    
    # Check for splatter condition
    if fall_counter > 20 and ground == 1:
        return expected_outputs  # All outputs are 0 if the Lemming splatters
    
    # Handle falling
    if ground == 0:
        if fall_counter <= 20:
            expected_outputs['aaah'] = 1
            fall_counter += 1
        else:
            expected_outputs['aaah'] = 1
            expected_outputs['walk_left'] = 0
            expected_outputs['walk_right'] = 0
            expected_outputs['digging'] = 0
            return expected_outputs  # Splatter after 20+ cycles of falling
    else:
        fall_counter = 0  # Reset fall counter when ground reappears
    
    # Handle digging
    if dig == 1 and ground == 1 and not (current_state['aaah'] or current_state['digging']):
        expected_outputs['digging'] = 1
        expected_outputs['walk_left'] = walking_left
        expected_outputs['walk_right'] = walking_right
    elif dig == 0 or ground == 0:
        expected_outputs['digging'] = 0
    
    # Handle bumping
    if bump_left == 1 and bump_right == 1:
        expected_outputs['walk_left'] = 0
        expected_outputs['walk_right'] = 1
    elif bump_left == 1:
        expected_outputs['walk_left'] = 0
        expected_outputs['walk_right'] = 1
    elif bump_right == 1:
        expected_outputs['walk_left'] = 1
        expected_outputs['walk_right'] = 0
    else:
        expected_outputs['walk_left'] = walking_left
        expected_outputs['walk_right'] = walking_right
    
    return expected_outputs, fall_counter
