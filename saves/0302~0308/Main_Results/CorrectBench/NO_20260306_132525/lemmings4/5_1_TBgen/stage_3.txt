import numpy as np

# Define the states
WALK_LEFT = 0b0001
WALK_RIGHT = 0b0010
FALLING = 0b0100
DIGGING = 0b1000
SPLATTERED = 0b10000

# Initialize the state and fall count
current_state = WALK_LEFT
fall_count = 0
previous_direction = WALK_LEFT

def update_state(bump_left, bump_right, ground, dig, areset):
    global current_state, fall_count, previous_direction
    
    if areset:
        # Asynchronous reset to WALK_LEFT
        current_state = WALK_LEFT
        fall_count = 0
        previous_direction = WALK_LEFT
        return (WALK_LEFT, 0, 0, 0)
    
    if current_state == SPLATTERED:
        # No transitions from SPLATTERED
        return (SPLATTERED, 0, 0, 0)
    
    if current_state == FALLING:
        if ground:
            if fall_count > 20:
                # Splatter if falling for more than 20 cycles
                current_state = SPLATTERED
                return (SPLATTERED, 0, 0, 0)
            else:
                # Resume walking in the previous direction
                current_state = previous_direction
                return (previous_direction, 0, 0, 0)
        else:
            # Continue falling
            fall_count += 1
            return (FALLING, 1, 0, 0)
    
    if current_state == DIGGING:
        if ground:
            # Resume walking in the previous direction
            current_state = previous_direction
            return (previous_direction, 0, 0, 0)
        else:
            # Transition to FALLING
            current_state = FALLING
            fall_count = 1
            return (FALLING, 1, 0, 0)
    
    if current_state == WALK_LEFT:
        if bump_left:
            # Switch to WALK_RIGHT
            current_state = WALK_RIGHT
            return (WALK_RIGHT, 0, 0, 0)
        elif bump_right:
            # Stay in WALK_LEFT
            return (WALK_LEFT, 0, 0, 0)
        elif not ground:
            # Transition to FALLING
            current_state = FALLING
            fall_count = 1
            return (FALLING, 1, 0, 0)
        elif dig and ground:
            # Transition to DIGGING
            current_state = DIGGING
            previous_direction = WALK_LEFT
            return (DIGGING, 0, 0, 1)
        else:
            # Stay in WALK_LEFT
            return (WALK_LEFT, 0, 0, 0)
    
    if current_state == WALK_RIGHT:
        if bump_left:
            # Stay in WALK_RIGHT
            return (WALK_RIGHT, 0, 0, 0)
        elif bump_right:
            # Switch to WALK_LEFT
            current_state = WALK_LEFT
            return (WALK_LEFT, 0, 0, 0)
        elif not ground:
            # Transition to FALLING
            current_state = FALLING
            fall_count = 1
            return (FALLING, 1, 0, 0)
        elif dig and ground:
            # Transition to DIGGING
            current_state = DIGGING
            previous_direction = WALK_RIGHT
            return (DIGGING, 0, 0, 1)
        else:
            # Stay in WALK_RIGHT
            return (WALK_RIGHT, 0, 0, 0)
