def update_state(bump_left, bump_right, ground, dig, areset):
    global current_state, fall_count, previous_direction

        # Asynchronous reset to WALK_LEFT
        current_state = WALK_LEFT
        fall_count = 0
        previous_direction = WALK_LEFT

        return (SPLATTERED, 0, 0, 0)

                # Splatter if falling for more than 20 cycles
                current_state = SPLATTERED
                return (SPLATTERED, 0, 0, 0)

                return (previous_direction, 0, 0, 0)
