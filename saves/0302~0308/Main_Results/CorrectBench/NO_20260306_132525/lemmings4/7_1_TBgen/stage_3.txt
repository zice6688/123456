import numpy as np

def lemming_fsm(bump_left, bump_right, ground, dig, current_state, fall_counter):
    # Define the states
    WALK_LEFT = 0b0001
    WALK_RIGHT = 0b0010
    FALLING = 0b0100
    DIGGING = 0b1000
    SPLATTERED = 0b0000

    # Initialize outputs
    walk_left = 0
    walk_right = 0
    aaah = 0
    digging = 0

    # Determine the next state and outputs based on the current state and inputs
    if current_state == SPLATTERED:
        # If the Lemming is splattered, it stays in that state
        pass
    elif current_state == WALK_LEFT:
        if bump_right:
            current_state = WALK_RIGHT
        elif bump_left:
            current_state = WALK_RIGHT
        elif not ground:
            current_state = FALLING
            fall_counter = 0
        elif dig:
            current_state = DIGGING
        else:
            walk_left = 1
    elif current_state == WALK_RIGHT:
        if bump_left:
            current_state = WALK_LEFT
        elif bump_right:
            current_state = WALK_LEFT
        elif not ground:
            current_state = FALLING
            fall_counter = 0
        elif dig:
            current_state = DIGGING
        else:
            walk_right = 1
    elif current_state == FALLING:
        if ground:
            if fall_counter > 20:
                current_state = SPLATTERED
            else:
                if current_state == WALK_LEFT:
                    current_state = WALK_LEFT
                elif current_state == WALK_RIGHT:
                    current_state = WALK_RIGHT
        else:
            fall_counter += 1
        aaah = 1
    elif current_state == DIGGING:
        if not ground:
            current_state = FALLING
            fall_counter = 0
        else:
            digging = 1

    # Return the next state and outputs
    return current_state, fall_counter, walk_left, walk_right, aaah, digging
