def lemming_rules(bump_left, bump_right, ground, dig, current_state, fall_counter):
    """

    Parameters:
    - bump_left (int): 1 if bumped from the left, 0 otherwise.
    - bump_right (int): 1 if bumped from the right, 0 otherwise.

    - dig (int): 1 if the Lemming should start digging, 0 otherwise.
    - current_state (dict): A dictionary containing the current state of the Lemming.
      - 'walk_left' (int): 1 if walking left, 0 otherwise.
      - 'walk_right' (int): 1 if walking right, 0 otherwise.
      - 'aaah' (int): 1 if falling, 0 otherwise.
      - 'digging' (int): 1 if digging, 0 otherwise.
    - fall_counter (int): Counter to track the number of clock cycles the Lemming has been falling.
    
    Returns:
    - new_state (dict): A dictionary containing the updated state of the Lemming.
    - new_fall_counter (int): Updated fall counter.
    """
    
    # Initialize the new state with the current state
    new_state = current_state.copy()
    new_fall_counter = fall_counter
    
    # If the Lemming is splattered, it remains in that state
    if current_state['walk_left'] == 0 and current_state['walk_right'] == 0 and current_state['aaah'] == 0 and current_state['digging'] == 0:
        return new_state, new_fall_counter

    if ground == 0:
        new_state['aaah'] = 1
        new_state['walk_left'] = 0
        new_state['walk_right'] = 0
        new_state['digging'] = 0
        new_fall_counter += 1
    else:
        if new_fall_counter > 20:
            # Splatter if the Lemming falls for more than 20 clock cycles and hits the ground
            new_state['walk_left'] = 0
            new_state['walk_right'] = 0
            new_state['aaah'] = 0
            new_state['digging'] = 0
            new_fall_counter = 0
        else:
            new_state['aaah'] = 0
            new_fall_counter = 0
    
    # Handle digging
    if ground == 1 and new_state['aaah'] == 0 and dig == 1:
        new_state['digging'] = 1
        new_state['walk_left'] = 0
        new_state['walk_right'] = 0
    elif ground == 0 and new_state['digging'] == 1:
        new_state['digging'] = 0
        new_state['aaah'] = 1
        new_state['walk_left'] = 0
        new_state['walk_right'] = 0
        new_fall_counter = 1
    else:
        new_state['digging'] = 0
    
    # Handle direction changes
    if new_state['aaah'] == 0 and new_state['digging'] == 0:
        if bump_left == 1:
            new_state['walk_left'] = 0
            new_state['walk_right'] = 1
