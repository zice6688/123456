import numpy as np

def lemming_fsm_rules(bump_left, bump_right, ground, dig, current_state, fall_counter):
    """
    This function implements the core rules of the Lemming FSM.
    
    Parameters:
    - bump_left: 1 if the Lemming is bumped on the left, 0 otherwise.
    - bump_right: 1 if the Lemming is bumped on the right, 0 otherwise.
    - ground: 1 if there is ground, 0 otherwise.
    - dig: 1 if the Lemming is told to dig, 0 otherwise.
    - current_state: A tuple (walk_left, walk_right, aaah, digging) representing the current state.
    - fall_counter: The number of clock cycles the Lemming has been falling.
    
    Returns:
    - next_state: A tuple (walk_left, walk_right, aaah, digging) representing the next state.
    - new_fall_counter: The updated fall counter.
    """
    walk_left, walk_right, aaah, digging = current_state
    
    # Initialize next state and new fall counter
    next_walk_left = walk_left
    next_walk_right = walk_right
    next_aaah = aaah
    next_digging = digging
    new_fall_counter = fall_counter
    
    # Handle reset condition
    if areset:
        next_walk_left = 1
        next_walk_right = 0
        next_aaah = 0
        next_digging = 0
        new_fall_counter = 0
        return (next_walk_left, next_walk_right, next_aaah, next_digging), new_fall_counter
    
    # Handle falling
    if not ground:
        if not aaah:
            next_aaah = 1
            new_fall_counter += 1
        next_walk_left = 0
        next_walk_right = 0
        next_digging = 0
    else:
        if aaah and new_fall_counter > 20:
            next_walk_left = 0
            next_walk_right = 0
            next_aaah = 0
            next_digging = 0
            new_fall_counter = 0
        else:
            next_aaah = 0
            new_fall_counter = 0
    
    # Handle digging
    if ground and not aaah and dig:
        next_digging = 1
        next_walk_left = 0
        next_walk_right = 0
    elif not ground and next_digging:
        next_digging = 0
        next_aaah = 1
        new_fall_counter = 1
    
    # Handle walking direction
    if bump_left and not bump_right and not aaah and not next_digging:
        next_walk_left = 0
        next_walk_right = 1
    elif bump_right and not bump_left and not aaah and not next_digging:
        next_walk_left = 1
        next_walk_right = 0
    elif bump_left and bump_right and not aaah and not next_digging:
        next_walk_left = 0 if walk_left else 1
        next_walk_right = 1 if walk_left else 0
    
    return (next_walk_left, next_walk_right, next_aaah, next_digging), new_fall_counter
