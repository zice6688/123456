# Define the states
WALK_LEFT = 0b0001
WALK_RIGHT = 0b0010
FALLING = 0b0100
DIGGING = 0b1000

# Initialize the state
current_state = WALK_LEFT
fall_counter = 0

def update_state(bump_left, bump_right, ground, dig):
    global current_state, fall_counter

    # Reset the fall counter if not falling
    if current_state != FALLING:
        fall_counter = 0

    # Determine the next state based on the current state and inputs
    if current_state == WALK_LEFT:
        if bump_left:
            current_state = WALK_RIGHT
        elif bump_right:
            current_state = WALK_LEFT
        elif not ground:
            current_state = FALLING
        elif dig:
            current_state = DIGGING
    elif current_state == WALK_RIGHT:
        if bump_left:
            current_state = WALK_RIGHT
        elif bump_right:
            current_state = WALK_LEFT
        elif not ground:
            current_state = FALLING
        elif dig:
            current_state = DIGGING
    elif current_state == FALLING:
        if ground:
            if fall_counter > 20:
                current_state = SPLATTERED
            else:
                if current_state == FALLING and (bump_left or bump_right):
                    current_state = WALK_LEFT if bump_right else WALK_RIGHT

                    current_state = WALK_LEFT if (current_state == WALK_LEFT) else WALK_RIGHT
        else:
            fall_counter += 1
    elif current_state == DIGGING:

                current_state = DIGGING
            else:
                current_state = WALK_LEFT if (current_state == WALK_LEFT) else WALK_RIGHT

            current_state = FALLING
