import numpy as np

def lemming_rules(bump_left, bump_right, ground, dig, current_state, fall_counter):
    # Define the states
    WALK_LEFT = 0b0001
    WALK_RIGHT = 0b0010
    FALLING = 0b0100
    DIGGING = 0b1000
    SPLATTERED = 0b0000

    # Initialize the next state and outputs
    next_state = current_state
    walk_left = 0
    walk_right = 0
    aaah = 0
    digging = 0

    # Asynchronous reset: reset to WALK_LEFT
    if areset:
        next_state = WALK_LEFT
        walk_left = 1
        return next_state, walk_left, walk_right, aaah, digging, 0

    # Handle falling
    if not ground:
        if current_state == WALK_LEFT or current_state == WALK_RIGHT:
            next_state = FALLING
            aaah = 1
            fall_counter = 1
        elif current_state == FALLING:
            fall_counter += 1
            if fall_counter > 20:
                next_state = SPLATTERED
                fall_counter = 0
        elif current_state == DIGGING:
            next_state = FALLING
            aaah = 1
            fall_counter = 1
    else:
        if current_state == FALLING:
            if fall_counter > 20:
                next_state = SPLATTERED
                fall_counter = 0
            else:
                next_state = (WALK_LEFT if (current_state & WALK_LEFT) else WALK_RIGHT)
                if next_state == WALK_LEFT:
                    walk_left = 1
                else:
                    walk_right = 1
                fall_counter = 0
        elif current_state == DIGGING:
            next_state = (WALK_LEFT if (current_state & WALK_LEFT) else WALK_RIGHT)
            if next_state == WALK_LEFT:
                walk_left = 1
            else:
                walk_right = 1
            digging = 0

    # Handle digging
    if ground and dig and (current_state == WALK_LEFT or current_state == WALK_RIGHT):
        next_state = DIGGING
        digging = 1

    # Handle bumping
    if (bump_left or bump_right) and (current_state == WALK_LEFT or current_state == WALK_RIGHT):
        if bump_left:
            next_state = WALK_RIGHT
            walk_right = 1
        if bump_right:
            next_state = WALK_LEFT
            walk_left = 1

    # Return the next state and outputs
    return next_state, walk_left, walk_right, aaah, digging, fall_counter
