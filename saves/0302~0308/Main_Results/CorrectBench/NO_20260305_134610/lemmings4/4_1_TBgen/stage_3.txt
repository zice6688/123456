def lemming_rules(bump_left, bump_right, ground, dig, current_state, fall_counter):
    """
    This function implements the core rules for the Lemming's behavior based on the given inputs.
    
    Parameters:
    - bump_left: 1 if the Lemming is bumped on the left, 0 otherwise.
    - bump_right: 1 if the Lemming is bumped on the right, 0 otherwise.
    - ground: 1 if the Lemming is on the ground, 0 otherwise.
    - dig: 1 if the Lemming should start digging, 0 otherwise.
    - current_state: A tuple (walk_left, walk_right, aaah, digging) representing the current state of the Lemming.
    - fall_counter: The number of clock cycles the Lemming has been falling.
    
    Returns:
    - A tuple (new_state, new_fall_counter) where new_state is the new state of the Lemming and new_fall_counter is the updated fall counter.
    """
    walk_left, walk_right, aaah, digging = current_state
    
    # Initialize the new state to the current state
    new_walk_left, new_walk_right, new_aaah, new_digging = walk_left, walk_right, aaah, digging
    new_fall_counter = fall_counter
    
    # Check for reset conditions
    if bump_left == 1 and bump_right == 1:
        new_walk_left, new_walk_right, new_aaah, new_digging = 0, 0, 0, 0
    elif bump_left == 1:
        new_walk_left, new_walk_right = 0, 1
    elif bump_right == 1:
        new_walk_left, new_walk_right = 1, 0
    
    # Check for falling conditions
    if ground == 0:
        new_walk_left, new_walk_right, new_aaah, new_digging = 0, 0, 1, 0
        new_fall_counter += 1
    else:
        if new_fall_counter > 20:
            new_walk_left, new_walk_right, new_aaah, new_digging = 0, 0, 0, 0
            new_fall_counter = 0
        else:
            new_aaah = 0
            new_fall_counter = 0
            
            # Resume previous walking state
            if new_walk_left == 1 or new_walk_right == 1:
                pass
            else:
                if walk_left == 1:
                    new_walk_left = 1
                elif walk_right == 1:
                    new_walk_right = 1
    
    # Check for digging conditions
    if dig == 1 and ground == 1 and not (aaah == 1 or new_fall_counter > 0):
        new_walk_left, new_walk_right, new_digging = 0, 0, 1
    elif dig == 1 and ground == 0:
        new_walk_left, new_walk_right, new_aaah, new_digging = 0, 0, 1, 0
        new_fall_counter = 1
    elif dig == 0 and ground == 1 and new_digging == 1:
        if walk_left == 1:
            new_walk_left = 1
        elif walk_right == 1:
            new_walk_right = 1
        new_digging = 0
    
    return (new_walk_left, new_walk_right, new_aaah, new_digging), new_fall_counter
