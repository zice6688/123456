import numpy as np

def lemming_rules(bump_left, bump_right, ground, dig, current_state, fall_counter):
    """
    This function implements the core rules for the Lemming's behavior based on the given inputs.
    
    Parameters:
    - bump_left: 1 if the Lemming is bumped on the left, 0 otherwise.
    - bump_right: 1 if the Lemming is bumped on the right, 0 otherwise.
    - ground: 1 if the Lemming is on the ground, 0 otherwise.
    - dig: 1 if the Lemming is told to dig, 0 otherwise.
    - current_state: A tuple (walk_left, walk_right, aaah, digging) representing the current state of the Lemming.
    - fall_counter: The number of clock cycles the Lemming has been falling.
    
    Returns:
    - next_state: A tuple (walk_left, walk_right, aaah, digging) representing the next state of the Lemming.
    - next_fall_counter: The updated fall counter.
    """
    
    walk_left, walk_right, aaah, digging = current_state
    
    # Initialize next state and fall counter
    next_walk_left = walk_left
    next_walk_right = walk_right
    next_aaah = aaah
    next_digging = digging
    next_fall_counter = fall_counter
    
    # Reset conditions
    if areset:
        next_walk_left = 1
        next_walk_right = 0
        next_aaah = 0
        next_digging = 0
        next_fall_counter = 0
        return (next_walk_left, next_walk_right, next_aaah, next_digging), next_fall_counter
    
    # Ground check
    if ground == 0:
        # Falling
        next_aaah = 1
        next_walk_left = 0
        next_walk_right = 0
        next_digging = 0
        next_fall_counter += 1
    else:
        # On ground
        next_aaah = 0
        next_fall_counter = 0
        
        # Bump checks
        if bump_left == 1:
            next_walk_left = 0
            next_walk_right = 1
        elif bump_right == 1:
            next_walk_left = 1
            next_walk_right = 0
        
        # Digging check
        if dig == 1:
            next_digging = 1
            next_walk_left = 0
            next_walk_right = 0
        else:
            next_digging = 0
    
    # Splatter check
    if next_fall_counter > 20 and ground == 1:
        next_walk_left = 0
        next_walk_right = 0
        next_aaah = 0
        next_digging = 0
        next_fall_counter = 0
    
    return (next_walk_left, next_walk_right, next_aaah, next_digging), next_fall_counter
