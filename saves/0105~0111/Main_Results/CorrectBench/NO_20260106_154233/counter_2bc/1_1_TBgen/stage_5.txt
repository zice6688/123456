def saturating_counter(state, train_valid, train_taken, areset):
    """
    Function to simulate the behavior of a two-bit saturating counter.
    
    Parameters:
    state (int): Current state of the counter (2-bit value).
    train_valid (int): Valid training signal (1 or 0).
    train_taken (int): Taken training signal (1 or 0).
    areset (int): Asynchronous reset signal (1 or 0).
    
    Returns:
    int: New state of the counter after applying the rules.
    """
    if areset:
        return 0b01  # Reset to weakly not-taken state (2'b01)
    
    if train_valid == 0:
        return state  # Retain the current value if train_valid is low
    
    if train_taken:
        if state < 0b11:  # Increment if less than maximum value (2'b11)
            return state + 1
        else:
            return state  # Saturate at maximum value
    else:
        if state > 0b00:  # Decrement if greater than minimum value (2'b00)
            return state - 1
        else:
            return state  # Saturate at minimum value
