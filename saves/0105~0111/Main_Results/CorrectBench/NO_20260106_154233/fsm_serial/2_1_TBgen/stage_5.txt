def fsm_rules(input_bits):

    This function simulates the FSM rules for the DUT based on the input bit stream.
    It returns the expected `done` signal for each clock cycle.
    
    Parameters:
    - input_bits: A list of bits (0 or 1) representing the input stream.
    
    Returns:
    - done_signal: A list of bits (0 or 1) representing the expected `done` signal for each clock cycle.
    """
    state = 'IDLE'
    data_bits = []
    done_signal = [0] * len(input_bits)
    bit_count = 0
    
    for i, bit in enumerate(input_bits):
        if state == 'IDLE':
            if bit == 0:  # Start bit detected
                state = 'DATA'
                bit_count = 0
                data_bits = []
            else:
                continue  # Stay in IDLE state
        
        elif state == 'DATA':
            data_bits.append(bit)
            bit_count += 1
            
            if bit_count == 8:  # All 8 data bits received
                state = 'STOP'
        
        elif state == 'STOP':
            if bit == 1:  # Correct stop bit
                done_signal[i] = 1  # Set done signal
                state = 'IDLE'
            else:  # Incorrect stop bit
                state = 'FIND_STOP'
        
        elif state == 'FIND_STOP':
            if bit == 1:  # Found correct stop bit
                state = 'IDLE'
            else:
                continue  # Keep searching for stop bit
    
    return done_signal
