import numpy as np

def is_message_start(byte):
    return (byte & 0x10) == 0x10

def generate_expected_values(input_bytes, reset_signal):
    state = 'SEARCH'
    message_buffer = [0, 0, 0]
    expected_out_bytes = []
    expected_done = []

    for i, byte in enumerate(input_bytes):
        if reset_signal[i]:
            state = 'SEARCH'
            message_buffer = [0, 0, 0]
            expected_out_bytes.append(0)
            expected_done.append(0)
            continue

        if state == 'SEARCH':
            if is_message_start(byte):
                message_buffer[0] = byte
                state = 'RECEIVE_2ND_BYTE'
            expected_out_bytes.append(0)
            expected_done.append(0)

        elif state == 'RECEIVE_2ND_BYTE':
            message_buffer[1] = byte
            state = 'RECEIVE_3RD_BYTE'
            expected_out_bytes.append(0)
            expected_done.append(0)

        elif state == 'RECEIVE_3RD_BYTE':
            message_buffer[2] = byte
            out_bytes = (message_buffer[0] << 16) | (message_buffer[1] << 8) | message_buffer[2]
            expected_out_bytes.append(out_bytes)
            expected_done.append(1)
            state = 'SEARCH'

    return np.array(expected_out_bytes), np.array(expected_done)
