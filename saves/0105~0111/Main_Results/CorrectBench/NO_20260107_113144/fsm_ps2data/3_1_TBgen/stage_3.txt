def fsm_rules(in_bytes, reset):
    # Initialize the state and output variables
    state = 'IDLE'
    out_bytes = 0x000000
    done = 0
    message_buffer = [0, 0, 0]
    message_count = 0

    for byte in in_bytes:
        if reset:
            state = 'IDLE'
            out_bytes = 0x000000
            done = 0
            message_buffer = [0, 0, 0]
            message_count = 0
            continue

        if state == 'IDLE':
            if (byte & 0b10000) == 0b10000:  # Check if in[3] = 1
                state = 'RECEIVING'
                message_buffer[0] = byte
                message_count = 1
            else:
                continue
        elif state == 'RECEIVING':
            if message_count == 1:
                message_buffer[1] = byte
                message_count += 1
            elif message_count == 2:
                message_buffer[2] = byte
                message_count += 1
                out_bytes = (message_buffer[0] << 16) | (message_buffer[1] << 8) | message_buffer[2]
                done = 1
                state = 'IDLE'
            else:
                continue

    return out_bytes, done
