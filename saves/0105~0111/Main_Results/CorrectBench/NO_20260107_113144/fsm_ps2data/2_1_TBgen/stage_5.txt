import numpy as np

def is_message_start(byte):
    """Check if the byte is the start of a message (in[3] == 1)."""
    return (byte & 0b10000) != 0

def process_input_stream(input_stream, reset_signal):
    """
    Process the input stream and generate the expected 'done' signal and 'out_bytes' output.
    
    :param input_stream: List of bytes (integers) representing the input stream.
    :param reset_signal: List of boolean values indicating when the reset signal is active.
    :return: A tuple (done_signal, out_bytes) where:
             - done_signal is a list of boolean values indicating when the 'done' signal should be asserted.
             - out_bytes is a list of 24-bit integers representing the expected 'out_bytes' output.
    """
    state = 'IDLE'
    message_bytes = []
    done_signal = []
    out_bytes = []

    for i, byte in enumerate(input_stream):
        if reset_signal[i]:
            state = 'IDLE'
            message_bytes = []
        
        if state == 'IDLE':
            if is_message_start(byte):
                state = 'RECEIVING'
                message_bytes.append(byte)
        elif state == 'RECEIVING':
            message_bytes.append(byte)
            if len(message_bytes) == 3:
                state = 'DONE'
                done_signal.append(True)
                out_bytes.append((message_bytes[0] << 16) | (message_bytes[1] << 8) | message_bytes[2])
                message_bytes = []
                state = 'IDLE'
            else:
                done_signal.append(False)
                out_bytes.append(0x000000)  # Don't-care value
        else:
            done_signal.append(False)
            out_bytes.append(0x000000)  # Don't-care value

    return done_signal, out_bytes
