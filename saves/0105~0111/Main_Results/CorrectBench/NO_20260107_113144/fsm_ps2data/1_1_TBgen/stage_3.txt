def is_start_of_message(byte):
    # Check if the 4th bit (in[3]) of the byte is 1
    return (byte & 0b00010000) != 0

def process_byte_stream(byte_stream, reset_signal):
    state = 'IDLE'
    message_bytes = [None, None, None]
    done_signals = []
    out_bytes = []

    for i, byte in enumerate(byte_stream):
        if reset_signal[i]:
            state = 'IDLE'
            message_bytes = [None, None, None]
            done_signals.append(0)
            out_bytes.append(None)
            continue

        if state == 'IDLE':
            if is_start_of_message(byte):
                state = 'COLLECTING'
                message_bytes[0] = byte
                done_signals.append(0)
                out_bytes.append(None)
            else:
                done_signals.append(0)
                out_bytes.append(None)
        elif state == 'COLLECTING':
            if message_bytes[1] is None:
                message_bytes[1] = byte
                done_signals.append(0)
                out_bytes.append(None)
            elif message_bytes[2] is None:
                message_bytes[2] = byte

                done_signals.append(1)
                out_bytes.append((message_bytes[0] << 16) | (message_bytes[1] << 8) | message_bytes[2])

                state = 'IDLE'

                out_bytes.append(None)

    return done_signals, out_bytes
