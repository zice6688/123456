import numpy as np

def lemming_fsm(bump_left, bump_right, ground, dig, current_state, fall_counter):
    """
    This function implements the core rules of the Lemming FSM.
    
    Parameters:
    - bump_left (int): 1 if the Lemming is bumped on the left, 0 otherwise.
    - bump_right (int): 1 if the Lemming is bumped on the right, 0 otherwise.
    - ground (int): 1 if there is ground, 0 otherwise.
    - dig (int): 1 if the Lemming should start digging, 0 otherwise.
    - current_state (str): The current state of the Lemming. Possible values: 'walk_left', 'walk_right', 'fall', 'dig'.
    - fall_counter (int): Counter for the number of clock cycles the Lemming has been falling.
    
    Returns:
    - next_state (str): The next state of the Lemming.
    - outputs (dict): The output signals of the Lemming FSM.
    - new_fall_counter (int): The updated fall counter.
    """
    # Initialize outputs
    walk_left = 0
    walk_right = 0
    aaah = 0
    digging = 0
    
    # Determine the next state and outputs based on the current state and inputs
    if current_state == 'walk_left':
        if bump_left == 1 or (bump_left == 1 and bump_right == 1):
            next_state = 'walk_right'
        elif bump_right == 1:
            next_state = 'walk_left'
        elif ground == 0:
            next_state = 'fall'
            aaah = 1
        elif dig == 1 and ground == 1:
            next_state = 'dig'
            digging = 1
        else:
            next_state = 'walk_left'
            walk_left = 1
    elif current_state == 'walk_right':
        if bump_right == 1 or (bump_left == 1 and bump_right == 1):
            next_state = 'walk_left'
        elif bump_left == 1:
            next_state = 'walk_right'
        elif ground == 0:
            next_state = 'fall'
            aaah = 1
        elif dig == 1 and ground == 1:
            next_state = 'dig'
            digging = 1
        else:
            next_state = 'walk_right'
            walk_right = 1
    elif current_state == 'fall':
        if ground == 1:
            if fall_counter >= 20:
                next_state = 'splatter'
            else:
                next_state = 'walk_left' if current_state == 'walk_left' else 'walk_right'
                walk_left = 1 if current_state == 'walk_left' else 0
                walk_right = 1 if current_state == 'walk_right' else 0
        else:
            next_state = 'fall'
            aaah = 1
        new_fall_counter = fall_counter + 1
    elif current_state == 'dig':
        if ground == 0:
            next_state = 'fall'
            aaah = 1
        else:
            next_state = 'dig'
            digging = 1
    elif current_state == 'splatter':
        next_state = 'splatter'
    else:
        raise ValueError("Invalid current state")
    
    # Update fall counter
    if next_state == 'fall':
        new_fall_counter = fall_counter + 1
    else:
        new_fall_counter = 0
    
    # Return the next state, outputs, and new fall counter
    return next_state, {'walk_left': walk_left, 'walk_right': walk_right, 'aaah': aaah, 'digging': digging}, new_fall_counter
