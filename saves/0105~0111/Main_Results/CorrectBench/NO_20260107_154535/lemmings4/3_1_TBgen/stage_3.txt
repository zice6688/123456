import numpy as np

def lemming_fsm(bump_left, bump_right, ground, dig, current_state, fall_counter):
    """
    This function simulates the behavior of the Lemming FSM based on the given inputs.
    
    Parameters:
    - bump_left (int): 1 if bumped from the left, 0 otherwise.
    - bump_right (int): 1 if bumped from the right, 0 otherwise.
    - ground (int): 1 if there is ground, 0 otherwise.
    - dig (int): 1 if the Lemming should start digging, 0 otherwise.
    - current_state (dict): A dictionary containing the current state of the Lemming.
                            Keys: 'walk_left', 'walk_right', 'aaah', 'digging'
    - fall_counter (int): A counter to track the number of clock cycles the Lemming has been falling.
    
    Returns:
    - new_state (dict): A dictionary containing the new state of the Lemming.
                        Keys: 'walk_left', 'walk_right', 'aaah', 'digging'
    - new_fall_counter (int): The updated fall counter.
    """
    
    # Extract current state
    walk_left = current_state['walk_left']
    walk_right = current_state['walk_right']
    aaah = current_state['aaah']
    digging = current_state['digging']
    
    # Initialize new state and fall counter
    new_state = {
        'walk_left': 0,
        'walk_right': 0,
        'aaah': 0,
        'digging': 0
    }
    new_fall_counter = fall_counter
    
    # Check for splatter condition
    if fall_counter > 20 and ground == 1:
        return new_state, 0  # Splatter, reset all outputs to 0
    
    # Check for falling condition
    if ground == 0:
        new_state['aaah'] = 1
        new_fall_counter += 1
        return new_state, new_fall_counter
    
    # Check for digging condition
    if ground == 1 and not (aaah or digging):
        if dig == 1:
            new_state['digging'] = 1
            return new_state, 0  # Reset fall counter
        else:
            new_state['digging'] = 0
    
    # Check for walking direction change
    if bump_left == 1 and bump_right == 1:
        if walk_left == 1:
            new_state['walk_right'] = 1
        elif walk_right == 1:
            new_state['walk_left'] = 1
    elif bump_left == 1:
        if walk_left == 1:
            new_state['walk_right'] = 1
        elif walk_right == 1:
            new_state['walk_right'] = 1
    elif bump_right == 1:
        if walk_left == 1:
            new_state['walk_left'] = 1
        elif walk_right == 1:
            new_state['walk_left'] = 1
    else:
        if walk_left == 1:
            new_state['walk_left'] = 1
        elif walk_right == 1:
            new_state['walk_right'] = 1
    
    return new_state, new_fall_counter
