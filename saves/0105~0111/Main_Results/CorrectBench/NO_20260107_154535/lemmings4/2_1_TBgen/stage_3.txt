import numpy as np

def lemming_rules(bump_left, bump_right, ground, dig, current_state, fall_counter):
    """
    This function defines the core rules for the Lemming's behavior based on the inputs.
    
    Parameters:
    - bump_left (int): 1 if bumped on the left, 0 otherwise.
    - bump_right (int): 1 if bumped on the right, 0 otherwise.
    - ground (int): 1 if there is ground, 0 otherwise.
    - dig (int): 1 if the Lemming should start digging, 0 otherwise.
    - current_state (dict): A dictionary containing the current state of the Lemming.
                            Keys: 'walk_left', 'walk_right', 'aaah', 'digging'
    - fall_counter (int): Counter to track the number of clock cycles the Lemming has been falling.
    
    Returns:
    - new_state (dict): A dictionary containing the updated state of the Lemming.
                        Keys: 'walk_left', 'walk_right', 'aaah', 'digging'
    - new_fall_counter (int): Updated fall counter.
    - splattered (bool): True if the Lemming has splattered, False otherwise.
    """
    
    # Initialize the new state with the current state
    new_state = current_state.copy()
    new_fall_counter = fall_counter
    splattered = False
    
    # If the Lemming is splattered, all outputs are 0
    if current_state['walk_left'] == 0 and current_state['walk_right'] == 0 and current_state['aaah'] == 0 and current_state['digging'] == 0:
        return new_state, new_fall_counter, True
    
    # Check for reset conditions
    if areset:
        new_state = {'walk_left': 1, 'walk_right': 0, 'aaah': 0, 'digging': 0}
        new_fall_counter = 0
        return new_state, new_fall_counter, False
    
    # Check for ground conditions
    if ground == 0:
        # Lemming is falling
        new_state['aaah'] = 1
        new_state['walk_left'] = 0
        new_state['walk_right'] = 0
        new_state['digging'] = 0
        new_fall_counter += 1
        if new_fall_counter > 20 and ground == 1:
            # Lemming splatters after falling for more than 20 cycles and hitting the ground
            new_state = {'walk_left': 0, 'walk_right': 0, 'aaah': 0, 'digging': 0}
            splattered = True
    else:
        # Lemming is not falling
        new_state['aaah'] = 0
        new_fall_counter = 0
        
        # Check for bump conditions
        if bump_left == 1 or bump_right == 1:
            if bump_left == 1:
                new_state['walk_left'] = 0
                new_state['walk_right'] = 1
            elif bump_right == 1:
                new_state['walk_left'] = 1
                new_state['walk_right'] = 0
            new_state['digging'] = 0  # Bumping stops digging
        else:
            # No bumps, check for digging
            if dig == 1 and current_state['digging'] == 0:
                new_state['digging'] = 1
                new_state['walk_left'] = 0
                new_state['walk_right'] = 0
            elif dig == 0 and current_state['digging'] == 1:
                new_state['digging'] = 0
                new_state['walk_left'] = current_state['walk_left']
                new_state['walk_right'] = current_state['walk_right']
    
    return new_state, new_fall_counter, splattered
