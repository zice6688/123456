In the Lemmings' 2D world, Lemmings can be in one of two states: walking left (walk_left is 1) or walking right (walk_right is 1). It will switch directions if it hits an obstacle. In particular, if a Lemming is bumped on the left (by receiving a 1 on bump_left), it will walk right. If it's bumped on the right (by receiving a 1 on bump_right), it will walk left. Being bumped while falling does not affect the walking direction, and being bumped in the same cycle as ground disappears (but not yet falling), or when the ground reappears while still falling, also does not affect the walking direction.
// In addition to walking and falling, Lemmings can sometimes be told to do useful things, like dig (it starts digging when dig=1). import numpy as np

def lemming_rules(bump_left, bump_right, ground, dig, current_state, fall_counter):
    # Define the states
    WALK_LEFT = 0b0001
    WALK_RIGHT = 0b0010
    FALLING = 0b0100
    DIGGING = 0b1000
    SPLATTERED = 0b10000

    # Initialize outputs
    walk_left = 0
    walk_right = 0
    aaah = 0
    digging = 0

    # Determine the next state and outputs based on the current state and inputs
    if current_state == WALK_LEFT:
        if bump_left:
            next_state = WALK_RIGHT
        elif bump_right:
            next_state = WALK_RIGHT
        elif not ground:
            next_state = FALLING
        elif dig:
            next_state = DIGGING
        else:
            next_state = WALK_LEFT
    elif current_state == WALK_RIGHT:
        if bump_left:
            next_state = WALK_LEFT
        elif bump_right:
            next_state = WALK_LEFT
        elif not ground:
            next_state = FALLING
        elif dig:
            next_state = DIGGING
        else:
            next_state = WALK_RIGHT
    elif current_state == FALLING:
        if ground:
            if fall_counter > 20:
                next_state = SPLATTERED
            else:
                next_state = current_state & (WALK_LEFT | WALK_RIGHT)
        else:
            next_state = FALLING
    elif current_state == DIGGING:
        if not ground:
            next_state = FALLING
        else:
            next_state = DIGGING
    elif current_state == SPLATTERED:
        next_state = SPLATTERED
    else:
        next_state = WALK_LEFT  # Default to walking left if in an unknown state

    # Set the outputs based on the next state
    if next_state == WALK_LEFT:
        walk_left = 1
    elif next_state == WALK_RIGHT:
        walk_right = 1
    elif next_state == FALLING:
        aaah = 1
    elif next_state == DIGGING:
        digging = 1
    elif next_state == SPLATTERED:
        pass  # All outputs are 0

    return next_state, walk_left, walk_right, aaah, digging
