import numpy as np

def lemming_rules(bump_left, bump_right, ground, dig, current_state, fall_counter):
    # Initial state
    walk_left = 1
    walk_right = 0
    aaah = 0
    digging = 0

    # Determine the current direction
    if current_state == 'walk_left':
        walk_left = 1
        walk_right = 0
    elif current_state == 'walk_right':
        walk_left = 0
        walk_right = 1
    elif current_state == 'falling':
        walk_left = 0
        walk_right = 0
        aaah = 1
    elif current_state == 'digging':
        walk_left = 0
        walk_right = 0
        digging = 1

    # Handle bumping
    if bump_left and not (current_state in ['falling', 'digging']):
        if current_state == 'walk_left':
            current_state = 'walk_right'
        elif current_state == 'walk_right':
            current_state = 'walk_left'
    elif bump_right and not (current_state in ['falling', 'digging']):
        if current_state == 'walk_left':
            current_state = 'walk_right'
        elif current_state == 'walk_right':
            current_state = 'walk_left'

    # Handle falling
    if ground == 0:
        if current_state in ['walk_left', 'walk_right']:
            current_state = 'falling'
            fall_counter = 0
        elif current_state == 'digging':
            current_state = 'falling'
            fall_counter = 0
    else:
        if current_state == 'falling':
            if fall_counter > 20:
                current_state = 'splattered'
            else:
                if current_state == 'walk_left':
                    current_state = 'walk_left'
                elif current_state == 'walk_right':
                    current_state = 'walk_right'

    # Handle digging
    if ground == 1 and not (current_state in ['falling', 'digging']) and dig:
        current_state = 'digging'
    elif current_state == 'digging' and ground == 0:
        current_state = 'falling'
        fall_counter = 0

    # Update outputs based on the current state
    if current_state == 'walk_left':
        walk_left = 1
        walk_right = 0
        aaah = 0
        digging = 0
    elif current_state == 'walk_right':
        walk_left = 0
        walk_right = 1
        aaah = 0
        digging = 0
    elif current_state == 'falling':
        walk_left = 0
        walk_right = 0
        aaah = 1
        digging = 0
    elif current_state == 'digging':
        walk_left = 0
        walk_right = 0
        aaah = 0
        digging = 1
    elif current_state == 'splattered':
        walk_left = 0
        walk_right = 0
        aaah = 0
        digging = 0

    return walk_left, walk_right, aaah, digging, current_state, fall_counter
