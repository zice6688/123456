def lemming_rules(current_state, bump_left, bump_right, ground, dig, fall_counter):
    """
    This function defines the core rules for the Lemming's behavior based on the given inputs.
    
    Parameters:
    - current_state: A 4-bit binary string representing the current state (walk_left, walk_right, aaah, digging)
    - bump_left: 1 if the Lemming is bumped on the left, 0 otherwise
    - bump_right: 1 if the Lemming is bumped on the right, 0 otherwise
    - ground: 1 if the ground is present, 0 otherwise
    - dig: 1 if the Lemming is told to dig, 0 otherwise
    - fall_counter: The number of clock cycles the Lemming has been falling
    
    Returns:
    - next_state: A 4-bit binary string representing the next state (walk_left, walk_right, aaah, digging)
    - new_fall_counter: The updated fall counter
    """
    
    # Unpack the current state
    walk_left, walk_right, aaah, digging = [int(bit) for bit in current_state]
    
    # Initialize the next state and new fall counter
    next_walk_left = walk_left
    next_walk_right = walk_right
    next_aaah = aaah
    next_digging = digging
    new_fall_counter = fall_counter
    
    # Handle asynchronous reset
    if areset:
        next_walk_left = 1
        next_walk_right = 0
        next_aaah = 0
        next_digging = 0
        new_fall_counter = 0
        return f"{next_walk_left}{next_walk_right}{next_aaah}{next_digging}", new_fall_counter
    
    # Handle falling
    if not ground:
        next_walk_left = 0
        next_walk_right = 0
        next_aaah = 1
        next_digging = 0
        new_fall_counter += 1
    else:
        if fall_counter > 20:
            # Splatter
            next_walk_left = 0
            next_walk_right = 0
            next_aaah = 0
            next_digging = 0
            new_fall_counter = 0
        else:
            # Resume walking or digging
            next_aaah = 0
            new_fall_counter = 0

                next_walk_left = 0
                next_walk_right = 0
                next_digging = 1
            else:
                next_digging = 0

                    next_walk_left = 0

                    next_walk_right = walk_right
    
    # Return the next state and new fall counter
    return f"{next_walk_left}{next_walk_right}{next_aaah}{next_digging}", new_fall_counter
