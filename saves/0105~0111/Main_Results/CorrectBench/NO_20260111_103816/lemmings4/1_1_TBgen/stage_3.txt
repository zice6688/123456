def lemming_rules(bump_left, bump_right, ground, dig, current_state, fall_counter):
    # Current state: [walk_left, walk_right, aaah, digging]
    walk_left, walk_right, aaah, digging = current_state
    
    # Initialize the next state to be the same as the current state
    next_state = current_state.copy()
    
    # Handle reset and initial state
    if areset:
        next_state = [1, 0, 0, 0]
        fall_counter = 0
        return next_state, fall_counter
    
    # Handle bumping
    if bump_left or bump_right:
        if bump_left and not bump_right:
            next_state[0], next_state[1] = 0, 1
        elif bump_right and not bump_left:
            next_state[0], next_state[1] = 1, 0
        elif bump_left and bump_right:
            next_state[0], next_state[1] = int(not walk_left), int(not walk_right)
    
    # Handle falling
    if not ground:
        next_state[2] = 1  # Start falling
        next_state[3] = 0  # Stop digging
        fall_counter += 1
    else:
        if aaah:
            if fall_counter > 20:
                next_state = [0, 0, 0, 0]  # Splatter
            else:
                next_state[2] = 0  # Stop falling
                next_state[3] = 0  # Stop digging
                next_state[0], next_state[1] = walk_left, walk_right  # Resume walking
            fall_counter = 0  # Reset fall counter when hitting the ground
    
    # Handle digging
    if ground and not aaah and dig:
        next_state[3] = 1  # Start digging
        next_state[0], next_state[1] = 0, 0  # Stop walking
    elif not ground and digging:
        next_state[3] = 0  # Stop digging
        next_state[2] = 1  # Start falling
    
    return next_state, fall_counter
