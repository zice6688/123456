# Define the states
WALK_LEFT = 0b0001

FALLING = 0b0100
DIGGING = 0b1000

# Define the outputs
def get_outputs(state):
    if state == WALK_LEFT:
        return {'walk_left': 1, 'walk_right': 0, 'aaah': 0, 'digging': 0}
    elif state == WALK_RIGHT:
        return {'walk_left': 0, 'walk_right': 1, 'aaah': 0, 'digging': 0}
    elif state == FALLING:
        return {'walk_left': 0, 'walk_right': 0, 'aaah': 1, 'digging': 0}
    elif state == DIGGING:
        return {'walk_left': 0, 'walk_right': 0, 'aaah': 0, 'digging': 1}
    elif state == SPLATTERED:
        return {'walk_left': 0, 'walk_right': 0, 'aaah': 0, 'digging': 0}

# Define the core rules for the DUT
def lemming_fsm(current_state, bump_left, bump_right, ground, dig, fall_count):
    next_state = current_state
    fall_count_next = fall_count
    outputs = get_outputs(current_state)

    if current_state == SPLATTERED:
        return SPLATTERED, fall_count_next, outputs

    if ground == 0:
        if current_state not in [FALLING, DIGGING]:
            next_state = FALLING
            fall_count_next = 1
        else:
            fall_count_next += 1
            if fall_count_next > 20 and ground == 1:
                next_state = SPLATTERED
                fall_count_next = 0

    return next_state, fall_count_next, outputs
