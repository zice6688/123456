import numpy as np

def lemming_fsm(inputs, current_state, fall_counter):
    """
    This function implements the core rules of the Lemming FSM.
    
    Parameters:
    - inputs: A dictionary containing the input signals (bump_left, bump_right, ground, dig).
    - current_state: A dictionary containing the current state (walk_left, walk_right, aaah, digging).
    - fall_counter: An integer representing the number of clock cycles the Lemming has been falling.
    
    Returns:
    - new_state: A dictionary containing the updated state (walk_left, walk_right, aaah, digging).
    - new_fall_counter: An integer representing the updated fall counter.
    """
    bump_left = inputs['bump_left']
    bump_right = inputs['bump_right']
    ground = inputs['ground']
    dig = inputs['dig']
    
    walk_left = current_state['walk_left']
    walk_right = current_state['walk_right']
    aaah = current_state['aaah']
    digging = current_state['digging']
    
    # Initialize new state and fall counter
    new_state = {
        'walk_left': 0,
        'walk_right': 0,
        'aaah': 0,
        'digging': 0
    }
    new_fall_counter = fall_counter
    
    # Check for reset or initial state
    if not (walk_left or walk_right or aaah or digging):
        new_state['walk_left'] = 1
        return new_state, new_fall_counter
    
    # Check for falling
    if not ground:
        new_state['aaah'] = 1
        new_fall_counter += 1
        return new_state, new_fall_counter
    
    # Check for splattering after a long fall
    if ground and new_fall_counter > 20:
        return new_state, new_fall_counter
    
    # Reset fall counter if on ground
    if ground:
        new_fall_counter = 0
    
    # Check for bumping
    if bump_left and not bump_right:
        new_state['walk_right'] = 1
    elif bump_right and not bump_left:
        new_state['walk_left'] = 1
    elif bump_left and bump_right:
        if walk_left:
            new_state['walk_right'] = 1
        else:
            new_state['walk_left'] = 1
    
    # Check for digging
    if ground and not aaah and dig:
        new_state['digging'] = 1
        new_state['walk_left'] = 0
        new_state['walk_right'] = 0
    elif not ground and digging:
        new_state['aaah'] = 1
        new_state['digging'] = 0
    elif ground and digging:
        new_state['digging'] = 1
        new_state['walk_left'] = 0
        new_state['walk_right'] = 0

    if not (new_state['walk_left'] or new_state['walk_right'] or new_state['aaah'] or new_state['digging']):

            new_state['walk_left'] = 1
        elif walk_right:
            new_state['walk_right'] = 1
    
    return new_state, new_fall_counter
