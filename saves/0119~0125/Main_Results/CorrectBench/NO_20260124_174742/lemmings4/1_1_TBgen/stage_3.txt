def lemming_rules(bump_left, bump_right, ground, dig, current_state, fall_counter):
    """
    This function implements the core rules for the Lemming's behavior based on the given problem description.
    
    Parameters:
    - bump_left: 1 if bumped from the left, 0 otherwise
    - bump_right: 1 if bumped from the right, 0 otherwise
    - ground: 1 if on the ground, 0 otherwise
    - dig: 1 if digging command is active, 0 otherwise
    - current_state: A dictionary containing the current state of the Lemming (walk_left, walk_right, aaah, digging)
    - fall_counter: Counter to track the number of clock cycles the Lemming has been falling
    
    Returns:
    - next_state: A dictionary containing the next state of the Lemming (walk_left, walk_right, aaah, digging)
    - next_fall_counter: Updated fall counter
    """
    
    # Extract current state
    walk_left = current_state['walk_left']
    walk_right = current_state['walk_right']
    aaah = current_state['aaah']
    digging = current_state['digging']
    
    # Initialize next state and fall counter
    next_state = {
        'walk_left': walk_left,
        'walk_right': walk_right,
        'aaah': aaah,
        'digging': digging
    }
    next_fall_counter = fall_counter
    
    # Check for ground and falling
    if ground == 0:
        if not aaah:
            next_state['aaah'] = 1
            next_state['walk_left'] = 0
            next_state['walk_right'] = 0
            next_state['digging'] = 0
            next_fall_counter = 1
        else:
            next_fall_counter += 1
    else:
        if aaah:
            if next_fall_counter > 20:
                next_state['walk_left'] = 0
                next_state['walk_right'] = 0
                next_state['aaah'] = 0
                next_state['digging'] = 0
                next_fall_counter = 0
            else:
                next_state['aaah'] = 0
                next_state['walk_left'] = walk_left
                next_state['walk_right'] = walk_right
                next_state['digging'] = digging
                next_fall_counter = 0
    
    # Check for bumps
    if bump_left == 1 or bump_right == 1:
        if bump_left == 1 and bump_right == 1:
            if walk_left == 1:
                next_state['walk_left'] = 0
                next_state['walk_right'] = 1
            elif walk_right == 1:
                next_state['walk_left'] = 1
                next_state['walk_right'] = 0

            next_state['walk_left'] = 1

        next_state['digging'] = 1
        next_state['walk_left'] = 0
        next_state['walk_right'] = 0
    elif dig == 0 or ground == 0:
        next_state['digging'] = 0
    
    return next_state, next_fall_counter
