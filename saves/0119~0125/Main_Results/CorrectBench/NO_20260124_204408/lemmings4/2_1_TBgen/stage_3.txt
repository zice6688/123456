                new_intended_walk_left = 1
                new_intended_walk_right = 0
        # else: no bump → direction unchanged

    # --- STEP 7: Assign outputs based on current cycle's effective state

    if next_splattered:
        walk_left = 0
        walk_right = 0
        aaah = 0
        digging = 0
    elif will_fall:
        walk_left = 0
        walk_right = 0
        aaah = 1
        digging = 0
    elif is_digging_now:
        walk_left = 0
        walk_right = 0
        aaah = 0
        digging = 1
    else:
        # Walking state (ground==1, not falling, not digging, not splattered)
        # Outputs reflect intended direction
        walk_left = new_intended_walk_left
        walk_right = new_intended_walk_right
        aaah = 0
        digging = 0

    return (
        walk_left,
        walk_right,
        aaah,
        digging,
        next_splattered,
        next_fall_counter,
        new_intended_walk_left,
        new_intended_walk_right
    )


# Wrapper for direct use in testbench: given full state and inputs, returns outputs and next state.
# This is the core rule engine.
def lemming_oracle_step(
    # Current-cycle inputs (sampled at posedge clk, except areset handled asynchronously outside)
    bump_left: int,
    bump_right: int,
    ground: int,
    dig: int,
    # Previous-cycle state (golden model state maintained by testbench)
    state_intended_walk_left: int,   # 0 or 1
    state_intended_walk_right: int,  # 0 or 1
    state_aaah: int,                 # 0 or 1
    state_digging: int,              # 0 or 1
    state_splattered: bool,
    state_fall_counter: int,         # non-negative integer
):
    """
    Single-cycle oracle step.
    Returns: (walk_left, walk_right, aaah, digging, next_state_tuple)
    where next_state_tuple = (intended_walk_left, intended_walk_right, aaah, digging, splattered, fall_counter)
    """
    # Unpack for clarity
    prev_iwl = state_intended_walk_left
    prev_iwr = state_intended_walk_right
    prev_aaah = state_aaah
    prev_digging = state_digging
    prev_splattered = state_splattered
    prev_fc = state_fall_counter

    # Call core computation
    res = compute_expected_outputs(
        bump_left, bump_right, ground, dig,
        # These are *outputs* from previous cycle — but for intended direction, we use state vars
        prev_iwl, prev_iwr, prev_aaah, prev_digging,
        prev_splattered, prev_fc,
        prev_iwl, prev_iwr
    )

    wl, wr, aah, dig_out, nsplatter, nfc, niwl, niwr = res

    # Next state outputs mirror the logic: aaah and digging are outputs, so next state uses them
    # But note: aaah and digging in next state are *outputs* of this cycle, so they become state_aaah/state_digging next cycle
    next_state = (niwl, niwr, aah, dig_out, nsplatter, nfc)

    return (wl, wr, aah, dig_out, next_state)


# Initial state after reset (matches DUT reset behavior)
def lemming_reset_state():
    """
    Returns initial golden state tuple after asynchronous reset:
    (intended_walk_left, intended_walk_right, aaah, digging, splattered, fall_counter)
    """
    return (1, 0, 0, 0, False, 0)
