def compute_expected_outputs(
    t,  # current cycle index (0-based)
    history: dict,
    bump_left: int,
    bump_right: int,
    ground: int,
    dig: int,
    areset: int
):
    """
    Core behavioral rule engine for Lemming FSM.
    Returns tuple (walk_left, walk_right, aaah, digging) — all int (0 or 1).
    
    Inputs:
      - t: global simulation cycle index (used only for stateful tracking via history)
      - history: mutable dict holding persistent state across cycles:
          * 'state': str in {'walk_left', 'walk_right', 'falling', 'digging', 'splattered'}
          * 'pre_fall_dir': str in {'walk_left', 'walk_right'} — direction before fall started
          * 'pre_dig_dir': str in {'walk_left', 'walk_right'} — direction before dig started
          * 'fall_counter': int — counts consecutive cycles where aaah=1 (resets on exit from falling)
          * 'splattered': bool — latched splatter flag (set once, cleared only by areset)
      - bump_left, bump_right, ground, dig: current-cycle input values (0/1)
      - areset: current-cycle areset value (0/1) — asynchronous, so takes effect *immediately* this cycle
    
    Rules applied in precedence order (Fall > Dig > Bump), per problem spec.
    Moore outputs depend *only* on internal state at start of cycle (before input evaluation).
    All state updates occur *after* output generation (i.e., outputs reflect pre-update state).
    
    This function computes the *expected* outputs for cycle `t`, then updates `history` in-place
    to reflect state transitions that take effect *for next cycle* (i.e., registers update on clock edge).
    """
    
    # === ASYNCHRONOUS RESET HANDLING (takes immediate effect THIS cycle) ===
    if areset:
        # Reset overrides everything: force initial state
        history['state'] = 'walk_left'
        history['pre_fall_dir'] = 'walk_left'
        history['pre_dig_dir'] = 'walk_left'
        history['fall_counter'] = 0
        history['splattered'] = False
        return (1, 0, 0, 0)  # walk_left=1, others=0
    
    # === SPLATTERED STATE: IRREVERSIBLE (except by reset) ===
    if history.get('splattered', False):
        return (0, 0, 0, 0)
    
    # === READ CURRENT STATE FOR OUTPUT (Moore: outputs depend on *current* state) ===
    state = history.get('state', 'walk_left')
    walk_left = 1 if state == 'walk_left' else 0
    walk_right = 1 if state == 'walk_right' else 0
    aaah = 1 if state == 'falling' else 0
    digging = 1 if state == 'digging' else 0
    
    # === DETERMINE NEXT STATE (based on inputs and precedence: Fall > Dig > Bump) ===
    # Precedence hierarchy:
    #   (1) FALL: triggered if ground == 0 AND not splattered → forces 'falling'
    #   (2) DIG: allowed only if (ground == 1) AND (state in {'walk_left','walk_right'}) AND (dig == 1)
    #   (3) BUMP: allowed only if (ground == 1) AND (state in {'walk_left','walk_right'}) AND (not falling/digging)
    
    next_state = state
    next_pre_fall_dir = history.get('pre_fall_dir', 'walk_left')
    next_pre_dig_dir = history.get('pre_dig_dir', 'walk_left')
    next_fall_counter = history.get('fall_counter', 0)
    
    # --- (1) FALL PRECEDENCE ---
    if ground == 0:
        # Entering or continuing fall
        next_state = 'falling'
        # Preserve walking direction *at moment of fall entry* only if just entering
        if state not in ['falling', 'digging']:
            # Just entered fall from walking → record pre-fall direction
            next_pre_fall_dir = state
        # Fall counter increments *only while in falling state*
        next_fall_counter += 1
    else:
        # ground == 1: exiting fall or not in fall
        if state == 'falling':
            # Exiting fall → check splatter condition on *this rising edge*
            if next_fall_counter >= 21:
                # Splatter activated: latch forever (until reset)
                history['splattered'] = True
                return (0, 0, 0, 0)  # splatter takes effect *immediately this cycle*
            else:
                # Resume walking in pre_fall_dir
                next_state = next_pre_fall_dir
                next_fall_counter = 0  # reset counter on exit
        # If not falling, handle dig/bump below — but first, if currently digging and ground=1,
        # remain digging *only if dig still asserted*; otherwise may transition out.
    
    # --- (2) DIG PRECEDENCE (only applies if NOT falling AND ground==1 AND currently walking) ---

    if (ground == 1 and 
        state in ['walk_left', 'walk_right'] and 
        dig == 1 and 
        state != 'falling'):
        # Start digging *this cycle* (if not already digging)
        next_state = 'digging'
        next_pre_dig_dir = state  # record direction before dig
    
    # If already digging and ground stays 1, continue digging (no change)
    # If already digging and ground becomes 0 → handled above in FALL branch (so next_state becomes 'falling')

    # Only applies if: ground==1 AND state is walking (not falling/digging/splattered) AND no higher-precedence action occurred

    history['pre_fall_dir'] = next_pre_fall_dir
    history['pre_dig_dir'] = next_pre_dig_dir
    history['fall_counter'] = next_fall_counter
    
    return (walk_left, walk_right, aaah, digging)
