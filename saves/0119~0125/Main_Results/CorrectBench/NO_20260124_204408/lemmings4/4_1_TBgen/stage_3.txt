def compute_expected_outputs(
    # Inputs (scalar per cycle — no arrays; called once per test vector)
    bump_left: int,           # 0 or 1
    bump_right: int,          # 0 or 1
    ground: int,              # 0 or 1
    dig: int,                 # 0 or 1
    # Internal state (carried across cycles — updated *after* computing outputs)
    prev_walk_left: int,      # 0 or 1 — direction before current cycle (Moore: determines next state's output)
    prev_walk_right: int,     # 0 or 1 — mutually exclusive with walk_left (but we track both for clarity; only one is 1 in valid states)
    prev_aaah: int,           # 0 or 1 — indicates current falling state (aaah=1 ⇔ falling)
    prev_digging: int,        # 0 or 1 — indicates current digging state
    fall_counter: int,        # non-negative int — counts consecutive cycles where ground==0 *and* not splattered
    splattered: bool,         # True iff splatter has occurred (irreversible until reset)
    # Note: 'last_direction_before_fall_or_dig' is *not* needed as separate var — it's encoded in (prev_walk_left, prev_walk_right)
    #   because Moore outputs are determined solely by current state, and walking direction persists through fall/dig
):
    """
    Core ideal behavioral rules for Lemming FSM.
    Returns tuple: (walk_left, walk_right, aaah, digging, new_fall_counter, new_splattered)
    All outputs are deterministic functions of inputs + internal state.
    Implements exact precedence: fall > dig > bump.
    Assumes synchronous update: outputs reflect *current* state; state updates happen *after* output evaluation.
    Does NOT handle areset — reset is external and forces full state reset (handled outside this function).
    """

    # --- PRECEDENCE LAYER 1: FALL (ground == 0) ---
    if ground == 0:
        # Falling starts or continues
        walk_left = 0
        walk_right = 0
        aaah = 1
        digging = 0
        new_fall_counter = fall_counter + 1
        new_splattered = splattered
    else:
        # ground == 1 → NOT falling *this cycle*, but may be resuming from fall/dig, or walking, or splattered
        if splattered:
            # Splattered is irreversible: all outputs zero forever
            walk_left = 0
            walk_right = 0
            aaah = 0
            digging = 0
            new_fall_counter = 0
            new_splattered = True
        elif prev_aaah == 1:
            # Was falling last cycle (so fall_counter >= 1), and ground just returned → check splatter
            if fall_counter > 20:
                # Splatter triggered *on this cycle* (ground=1 after >20-cycle fall)
                walk_left = 0
                walk_right = 0
                aaah = 0
                digging = 0
                new_fall_counter = 0
                new_splattered = True
            else:
                # Resume walking in *same direction as before the fall started*
                # Since direction is preserved in state, and prev_walk_* reflect pre-fall direction (because
                # during fall, walk_* stay 0 but direction memory is retained in state variables),
                # we use the *last non-zero walking direction* — which is encoded in prev_walk_left/prev_walk_right
                # from the most recent non-falling, non-digging, non-splattered state.
                # But note: during fall, prev_walk_* remain unchanged (they’re state-holding), so they *are* the memory.
                walk_left = prev_walk_left
                walk_right = prev_walk_right
                aaah = 0
                digging = 0
                new_fall_counter = 0
                new_splattered = False
        elif prev_digging == 1:
            # Was digging last cycle, and ground=1 now → digging ends; resume walking in *same direction as before dig started*
            # Again, prev_walk_* hold that direction (digging doesn’t change them)
            walk_left = prev_walk_left
            walk_right = prev_walk_right
            aaah = 0
            digging = 0
            new_fall_counter = 0
            new_splattered = False
        else:
            # Not falling, not digging, not splattered → normal walking state. Now apply dig & bump logic.
            # --- PRECEDENCE LAYER 2: DIG (only if ground==1 AND not falling AND not splattered — already satisfied)
            if dig == 1:
                # Initiate or continue digging (digging persists until ground loss)
                walk_left = 0
                walk_right = 0
                aaah = 0
                digging = 1
                new_fall_counter = 0
                new_splattered = False
            else:
                # --- PRECEDENCE LAYER 3: DIRECTION SWITCH (only if NOT falling, NOT digging, NOT splattered, ground==1)

                curr_right = prev_walk_right

                    walk_left = 1
                    walk_right = 0

                else:
                    walk_left = curr_left
                    walk_right = curr_right

                aaah = 0
                digging = 0
                new_fall_counter = 0
                new_splattered = False

    # Final validation: ensure mutual exclusivity of walk_* (exactly one can be 1 in valid walking states; both 0 during fall/dig/splatter)
    # This is enforced by logic above.

    return (
        walk_left,
        walk_right,
        aaah,
        digging,
        new_fall_counter,
        new_splattered
    )
