import numpy as np

def lemming_rules(bump_left, bump_right, ground, dig, current_state, fall_counter):
    """
    This function defines the core rules for the Lemming's behavior based on the given inputs.
    
    Parameters:
    - bump_left: 1 if the Lemming is bumped on the left, 0 otherwise.
    - bump_right: 1 if the Lemming is bumped on the right, 0 otherwise.
    - ground: 1 if the Lemming is on the ground, 0 otherwise.
    - dig: 1 if the Lemming is told to dig, 0 otherwise.
    - current_state: A dictionary representing the current state of the Lemming.
        - 'walk_left': 1 if the Lemming is walking left, 0 otherwise.
        - 'walk_right': 1 if the Lemming is walking right, 0 otherwise.
        - 'aaah': 1 if the Lemming is falling, 0 otherwise.
        - 'digging': 1 if the Lemming is digging, 0 otherwise.
    - fall_counter: The number of clock cycles the Lemming has been falling.
    
    Returns:
    - next_state: A dictionary representing the next state of the Lemming.
    - next_fall_counter: The updated fall counter.
    """
    
    # Initialize the next state with the current state
    next_state = current_state.copy()
    next_fall_counter = fall_counter
    
    # Check if the Lemming is on the ground
    if ground == 1:
        # If the Lemming is on the ground, reset the fall counter
        next_fall_counter = 0
        
        # Check if the Lemming is digging
        if dig == 1 and (current_state['walk_left'] == 1 or current_state['walk_right'] == 1):
            next_state['digging'] = 1
            next_state['walk_left'] = 0
            next_state['walk_right'] = 0
            next_state['aaah'] = 0
        else:
            # If not digging, check for bumps
            if bump_left == 1:
                next_state['walk_left'] = 0
                next_state['walk_right'] = 1
                next_state['aaah'] = 0
                next_state['digging'] = 0
            elif bump_right == 1:
                next_state['walk_left'] = 1
                next_state['walk_right'] = 0
                next_state['aaah'] = 0
                next_state['digging'] = 0
            else:
                # No bumps, continue walking
                next_state['digging'] = 0
                next_state['aaah'] = 0
    else:
        # If the Lemming is not on the ground, it is falling
        next_state['aaah'] = 1
        next_state['walk_left'] = 0
        next_state['walk_right'] = 0
        next_state['digging'] = 0
        next_fall_counter += 1
        
        # Check if the Lemming has fallen for more than 20 clock cycles
        if next_fall_counter > 20:
            next_state['aaah'] = 0
            next_state['walk_left'] = 0
            next_state['walk_right'] = 0
            next_state['digging'] = 0
            next_fall_counter = 0  # Reset the fall counter after splattering
    
    return next_state, next_fall_counter
