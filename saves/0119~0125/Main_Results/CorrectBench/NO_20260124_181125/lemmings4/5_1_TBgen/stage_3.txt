def lemming_rules(bump_left, bump_right, ground, dig, current_state, fall_counter):
    """
    This function implements the core rules for the Lemming's behavior.
    
    Parameters:
    - bump_left: 1 if the Lemming is bumped on the left, 0 otherwise.
    - bump_right: 1 if the Lemming is bumped on the right, 0 otherwise.
    - ground: 1 if the Lemming is on the ground, 0 otherwise.
    - dig: 1 if the Lemming is digging, 0 otherwise.
    - current_state: A tuple (walk_left, walk_right, aaah, digging) representing the current state of the Lemming.
    - fall_counter: The number of clock cycles the Lemming has been falling.
    
    Returns:
    - next_state: A tuple (walk_left, walk_right, aaah, digging) representing the next state of the Lemming.
    - new_fall_counter: The updated fall counter.
    """
    walk_left, walk_right, aaah, digging = current_state
    
    # Initialize the next state to the current state
    next_state = (walk_left, walk_right, aaah, digging)
    new_fall_counter = fall_counter
    
    # Rule 1: Asynchronous reset
    if areset:
        next_state = (1, 0, 0, 0)
        new_fall_counter = 0
        return next_state, new_fall_counter
    
    # Rule 2: Switch directions when bumped
    if bump_left or bump_right:
        if bump_left and not bump_right:
            next_state = (0, 1, 0, 0)
        elif bump_right and not bump_left:
            next_state = (1, 0, 0, 0)
        else:  # Both bump_left and bump_right are 1
            if walk_left:
                next_state = (0, 1, 0, 0)
            else:
                next_state = (1, 0, 0, 0)
    
    # Rule 3: Falling
    if not ground:
        next_state = (0, 0, 1, 0)
        new_fall_counter += 1
    else:
        if new_fall_counter > 20:
            next_state = (0, 0, 0, 0)
            new_fall_counter = 0
        else:
            if walk_left:
                next_state = (1, 0, 0, 0)
            else:
                next_state = (0, 1, 0, 0)
            new_fall_counter = 0
    
    # Rule 4: Digging
    if ground and not aaah and dig:
        next_state = (0, 0, 0, 1)
    elif not ground and digging:
        next_state = (0, 0, 1, 0)
        new_fall_counter += 1
    elif ground and digging:
        next_state = (0, 0, 0, 0)
    
    return next_state, new_fall_counter
