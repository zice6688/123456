import numpy as np

def lemming_rules(bump_left, bump_right, ground, dig, current_state, fall_counter):
    """
    This function implements the core rules for the Lemming's behavior based on the given inputs.
    
    Parameters:
    - bump_left: 1 if the Lemming is bumped from the left, 0 otherwise.
    - bump_right: 1 if the Lemming is bumped from the right, 0 otherwise.
    - ground: 1 if the ground is present, 0 otherwise.
    - dig: 1 if the Lemming is told to start digging, 0 otherwise.
    - current_state: A dictionary containing the current state of the Lemming (walk_left, walk_right, aaah, digging).
    - fall_counter: The number of clock cycles the Lemming has been falling.
    
    Returns:
    - A dictionary with the updated state of the Lemming (walk_left, walk_right, aaah, digging).
    - The updated fall_counter.
    """
    # Unpack the current state
    walk_left = current_state['walk_left']
    walk_right = current_state['walk_right']
    aaah = current_state['aaah']
    digging = current_state['digging']

    # Initialize the new state
    new_state = {
        'walk_left': 0,
        'walk_right': 0,
        'aaah': 0,
        'digging': 0
    }

    # Handle splattering
    if fall_counter > 20 and ground == 1:
        return new_state, 0  # All outputs become 0, reset fall counter

    # Handle falling
    if ground == 0:
        new_state['aaah'] = 1
        fall_counter += 1
        return new_state, fall_counter

    # Reset fall counter if the Lemming hits the ground
    if ground == 1:
        fall_counter = 0

    # Handle digging
    if dig == 1 and ground == 1 and not aaah:
        new_state['digging'] = 1
        return new_state, fall_counter

    # Handle bumping
    if bump_left == 1 or bump_right == 1:
        if bump_left == 1:
            new_state['walk_right'] = 1
        elif bump_right == 1:
            new_state['walk_left'] = 1
        return new_state, fall_counter

    # Continue in the same direction if no other conditions are met
    if walk_left == 1:
        new_state['walk_left'] = 1
    elif walk_right == 1:
        new_state['walk_right'] = 1

    return new_state, fall_counter
