import numpy as np

def lemming_rules(bump_left, bump_right, ground, dig, current_state, fall_counter):
    """
    This function implements the core rules for the Lemming's behavior based on the given inputs.
    
    Parameters:
    - bump_left (int): 1 if the Lemming is bumped from the left, 0 otherwise.
    - bump_right (int): 1 if the Lemming is bumped from the right, 0 otherwise.
    - ground (int): 1 if there is ground, 0 if the Lemming is falling.
    - dig (int): 1 if the Lemming should start digging, 0 otherwise.
    - current_state (tuple): A tuple (walk_left, walk_right, aaah, digging) representing the current state.
    - fall_counter (int): The number of clock cycles the Lemming has been falling.
    
    Returns:
    - new_state (tuple): A tuple (walk_left, walk_right, aaah, digging) representing the new state.
    - new_fall_counter (int): The updated fall counter.
    """
    walk_left, walk_right, aaah, digging = current_state
    
    # Initialize new state and fall counter
    new_walk_left = walk_left
    new_walk_right = walk_right
    new_aaah = aaah
    new_digging = digging
    new_fall_counter = fall_counter
    
    # Reset conditions
    if not ground:
        new_aaah = 1
        new_digging = 0
        new_fall_counter += 1
    else:
        new_aaah = 0
        if new_fall_counter > 20:
            new_walk_left = 0
            new_walk_right = 0
            new_aaah = 0
            new_digging = 0
            new_fall_counter = 0
        else:
            new_fall_counter = 0
    
    # Bump conditions
    if not (aaah or digging):
        if bump_left and not bump_right:
            new_walk_left = 0
            new_walk_right = 1
        elif bump_right and not bump_left:
            new_walk_left = 1
            new_walk_right = 0
        elif bump_left and bump_right:
            new_walk_left = 0 if walk_left else 1
            new_walk_right = 1 if walk_left else 0
    
    # Digging conditions
    if ground and not (aaah or digging):
        if dig:
            new_digging = 1
            new_walk_left = 0
            new_walk_right = 0
        else:
            new_digging = 0
            if walk_left:
                new_walk_left = 1
                new_walk_right = 0
            else:
                new_walk_left = 0
                new_walk_right = 1
    
    return (new_walk_left, new_walk_right, new_aaah, new_digging), new_fall_counter
