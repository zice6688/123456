import numpy as np

def generate_expected_values(input_stream, reset_signal):
    # Initialize state variables
    state = 'IDLE'
    bit_count = 0
    data_bits = 0
    done_signal = [0] * len(input_stream)
    
    for i in range(len(input_stream)):
        if reset_signal[i]:
            state = 'IDLE'
            bit_count = 0
            data_bits = 0
            continue
        
        if state == 'IDLE':
            if input_stream[i] == 0:  # Start bit detected
                state = 'RECEIVING_DATA'
                bit_count = 0
                data_bits = 0
        elif state == 'RECEIVING_DATA':
            if bit_count < 8:
                data_bits = (data_bits << 1) | input_stream[i]
                bit_count += 1
            else:
                if input_stream[i] == 1:  # Stop bit detected
                    done_signal[i] = 1
                    state = 'IDLE'
                else:
                    state = 'SEARCHING_STOP_BIT'
        elif state == 'SEARCHING_STOP_BIT':
            if input_stream[i] == 1:  # Stop bit detected
                state = 'IDLE'
    
    return done_signal
