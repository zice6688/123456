module top_module (
    input clk,
    input in,
    input reset,
    output done
);

    typedef enum logic [2:0] {
        IDLE,
        START,
        DATA0,
        DATA1,
        DATA2,
        DATA3,
        DATA4,
        DATA5,
        DATA6,
        DATA7,
        STOP
    } state_t;

    state_t state, next_state;
    logic [7:0] data;
    logic [2:0] bit_count;

    always_ff @(posedge clk or posedge reset) begin
        if (reset) begin
            state <= IDLE;
            done <= 0;
            bit_count <= 0;
        end else begin
            state <= next_state;
            if (state == STOP && in == 1'b1) begin
                done <= 1;
            end else begin
                done <= 0;
            end
            if (state == DATA0 || state == DATA1 || state == DATA2 || state == DATA3 || state == DATA4 || state == DATA5 || state == DATA6 || state == DATA7) begin
                data[bit_count] <= in;
                bit_count <= bit_count + 1;
            end
        end
    end

    always_comb begin
        next_state = state;
        case (state)
            IDLE: begin
                if (in == 1'b0) begin
                    next_state = START;
                end
            end
            START: begin
                next_state = DATA0;
            end
            DATA0: begin
                next_state = DATA1;
            end
            DATA1: begin
                next_state = DATA2;
            end
            DATA2: begin
                next_state = DATA3;
            end
            DATA3: begin
                next_state = DATA4;
            end
            DATA4: begin
                next_state = DATA5;
            end
            DATA5: begin
                next_state = DATA6;
            end
            DATA6: begin
                next_state = DATA7;
            end
            DATA7: begin

            end
            STOP: begin

                    next_state = IDLE;
                end else begin
                    next_state = STOP; // Wait for the stop bit
                end
            end
        endcase
    end

endmodule
