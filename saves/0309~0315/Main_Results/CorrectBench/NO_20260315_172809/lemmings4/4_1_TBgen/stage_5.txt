import numpy as np

def initialize_dut(areset):
    """
    Initialize the DUT with a positive edge-triggered asynchronous reset.
    If areset is 1, the DUT should set walk_left = 1, walk_right = 0, aaah = 0, and digging = 0.
    """
    if areset == 1:
        return {'walk_left': 1, 'walk_right': 0, 'aaah': 0, 'digging': 0}
    else:
        return None

def bump_behavior(walk_left, walk_right, bump_left, bump_right):
    """
    Handle the behavior when the Lemming is bumped.
    - If walking left (walk_left = 1) and bumped on the left (bump_left = 1), switch to walking right.
    - If walking right (walk_right = 1) and bumped on the right (bump_right = 1), switch to walking left.
    """
    if walk_left == 1 and bump_left == 1:
        return {'walk_left': 0, 'walk_right': 1, 'aaah': 0, 'digging': 0}
    elif walk_right == 1 and bump_right == 1:
        return {'walk_left': 1, 'walk_right': 0, 'aaah': 0, 'digging': 0}
    else:
        return {'walk_left': walk_left, 'walk_right': walk_right, 'aaah': 0, 'digging': 0}

def ground_behavior(walk_left, walk_right, ground, falling_counter):
    """
    Handle the behavior when the ground disappears or reappears.
    - If the ground disappears (ground = 0), the Lemming will fall and say 'aaah!' (aaah = 1).
    - When the ground reappears (ground = 1), the Lemming will resume walking in the same direction as before the fall.
    - If the Lemming falls for more than 20 clock cycles and then hits the ground, it will splatter and all outputs will become 0.
    """
    if ground == 0:
        if falling_counter < 20:
            return {'walk_left': 0, 'walk_right': 0, 'aaah': 1, 'digging': 0}, falling_counter + 1
        else:
            return {'walk_left': 0, 'walk_right': 0, 'aaah': 0, 'digging': 0}, falling_counter + 1
    else:
        if falling_counter >= 20:
            return {'walk_left': 0, 'walk_right': 0, 'aaah': 0, 'digging': 0}, 0
        else:
            if walk_left == 1:
                return {'walk_left': 1, 'walk_right': 0, 'aaah': 0, 'digging': 0}, 0
            elif walk_right == 1:
                return {'walk_left': 0, 'walk_right': 1, 'aaah': 0, 'digging': 0}, 0
            else:
                return {'walk_left': 0, 'walk_right': 0, 'aaah': 0, 'digging': 0}, 0

def dig_behavior(walk_left, walk_right, ground, dig, digging):
    """
    Handle the behavior when the Lemming is told to dig.
    - If the Lemming is currently walking on the ground (ground = 1 and not falling), and the dig signal is asserted (dig = 1),
      the Lemming will start digging (digging = 1) and continue until the ground disappears (ground = 0).
    - At that point, since there is no ground, it will fall (aaah = 1), then continue walking in its original direction once it hits the ground again.
    """
    if ground == 1 and dig == 1 and not digging:
        return {'walk_left': 0, 'walk_right': 0, 'aaah': 0, 'digging': 1}
    elif ground == 0 and digging:
        return {'walk_left': 0, 'walk_right': 0, 'aaah': 1, 'digging': 0}
    else:
        return {'walk_left': walk_left, 'walk_right': walk_right, 'aaah': 0, 'digging': digging}

def update_state(walk_left, walk_right, aaah, digging, bump_left, bump_right, ground, dig, falling_counter):
    """
    Update the state of the DUT based on the current inputs and the rules defined above.
    """
    # Initialize the new state
    new_state = {'walk_left': walk_left, 'walk_right': walk_right, 'aaah': aaah, 'digging': digging}
    
    # Apply bump behavior
    bump_result = bump_behavior(walk_left, walk_right, bump_left, bump_right)
    new_state.update(bump_result)
    
    # Apply ground behavior

    new_state.update(ground_result)

        dig_result = dig_behavior(new_state['walk_left'], new_state['walk_right'], ground, dig, new_state['digging'])
        new_state.update(dig_result)
    
    return new_state, new_falling_counter
