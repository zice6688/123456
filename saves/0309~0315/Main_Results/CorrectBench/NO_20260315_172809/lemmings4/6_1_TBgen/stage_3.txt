import numpy as np

def lemming_rules(bump_left, bump_right, ground, dig, fall_count, current_state):
    """
    This function implements the core rules for the Lemming's behavior.
    
    Parameters:
    - bump_left: 1 if the Lemming is bumped on the left, 0 otherwise.
    - bump_right: 1 if the Lemming is bumped on the right, 0 otherwise.
    - ground: 1 if the ground is present, 0 otherwise.
    - dig: 1 if the Lemming is told to dig, 0 otherwise.
    - fall_count: Number of clock cycles the Lemming has been falling.
    - current_state: Current state of the Lemming (walk_left, walk_right, aaah, digging).
    
    Returns:
    - next_state: The next state of the Lemming (walk_left, walk_right, aaah, digging).
    - new_fall_count: Updated fall count.
    """
    walk_left, walk_right, aaah, digging = current_state
    
    # Initialize next state and new fall count
    next_state = [walk_left, walk_right, aaah, digging]
    new_fall_count = fall_count
    
    # Check for splatter condition
    if fall_count > 20 and ground == 1:
        next_state = [0, 0, 0, 0]
        new_fall_count = 0
        return next_state, new_fall_count
    
    # Check for falling
    if ground == 0:
        next_state = [0, 0, 1, 0]
        new_fall_count += 1
        return next_state, new_fall_count
    
    # Reset fall count if not falling
    if ground == 1:
        new_fall_count = 0
    
    # Check for digging
    if ground == 1 and dig == 1:
        next_state = [0, 0, 0, 1]
        return next_state, new_fall_count
    
    # Check for bump conditions
    if bump_left == 1 or bump_right == 1:
        if bump_left == 1:
            next_state = [0, 1, 0, 0]
        elif bump_right == 1:
            next_state = [1, 0, 0, 0]
        else:
            next_state = [not walk_left, not walk_right, 0, 0]
    
    # Resume walking in the same direction after falling
    if ground == 1 and (walk_left == 1 or walk_right == 1):
        next_state = [walk_left, walk_right, 0, 0]
    
    return next_state, new_fall_count
