import numpy as np

# Define the states
WL = 0b0001  # Walking Left
WR = 0b0010  # Walking Right
F = 0b0100   # Falling
D = 0b1000   # Digging
S = 0b10000  # Splattered

# Define the outputs
WALK_LEFT = 0b0001
WALK_RIGHT = 0b0010
AAAH = 0b0100
DIGGING = 0b1000

def get_expected_outputs(current_state, bump_left, bump_right, ground, dig, fall_duration):
    if current_state == WL:
        if bump_left:
            return WR, WALK_RIGHT
        elif not ground:
            return F, AAAH
        elif dig and ground:
            return D, DIGGING
        else:
            return WL, WALK_LEFT
    elif current_state == WR:
        if bump_right:
            return WL, WALK_LEFT
        elif not ground:
            return F, AAAH
        elif dig and ground:
            return D, DIGGING
        else:
            return WR, WALK_RIGHT
    elif current_state == F:
        if ground and fall_duration > 20:
            return S, 0
        elif ground and fall_duration <= 20:
            if current_state & WL:
                return WL, WALK_LEFT
            else:
                return WR, WALK_RIGHT
        else:
            return F, AAAH
    elif current_state == D:
        if not ground:
            return F, AAAH
        elif ground and not dig:
            if current_state & WL:
                return WL, WALK_LEFT
            else:
                return WR, WALK_RIGHT
        else:
            return D, DIGGING
    elif current_state == S:
        return S, 0
    else:
        raise ValueError("Unknown state")

# Example usage
current_state = WL
bump_left = 0
bump_right = 0
ground = 1
dig = 0
fall_duration = 0

next_state, expected_output = get_expected_outputs(current_state, bump_left, bump_right, ground, dig, fall_duration)
print(f"Next State: {bin(next_state)}, Expected Output: {bin(expected_output)}")
