import numpy as np

# Define the states
WL = 0b0001  # Walking Left
WR = 0b0010  # Walking Right
F = 0b0100   # Falling
D = 0b1000   # Digging
S = 0b0000   # Splattered

# Function to determine the next state and outputs based on the current state and inputs
def lemming_fsm(current_state, bump_left, bump_right, ground, dig, fall_count):
    next_state = current_state
    walk_left = 0
    walk_right = 0
    aaah = 0
    digging = 0
    
    if current_state == WL:
        if bump_left:
            next_state = WR
        elif not ground:
            next_state = F
            fall_count = 0
        elif dig and ground:
            next_state = D
        else:
            walk_left = 1
    elif current_state == WR:
        if bump_right:
            next_state = WL
        elif not ground:
            next_state = F
            fall_count = 0
        elif dig and ground:
            next_state = D
        else:
            walk_right = 1
    elif current_state == F:
        if ground:
            if fall_count >= 20:
                next_state = S
            else:
                if current_state == WL:
                    next_state = WL
                else:
                    next_state = WR
        else:
            fall_count += 1
        aaah = 1
    elif current_state == D:
        if not ground:
            next_state = F
            fall_count = 0
        elif ground:
            if current_state == WL:
                next_state = WL
            else:
                next_state = WR
        digging = 1
    elif current_state == S:
        pass  # Stay in splattered state
    
    return next_state, walk_left, walk_right, aaah, digging, fall_count
