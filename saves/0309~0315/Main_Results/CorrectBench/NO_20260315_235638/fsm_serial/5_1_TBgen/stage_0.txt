module top_module (
    input clk,
    input in,
    input reset,
    output reg done
);

    typedef enum logic [2:0] {
        IDLE,
        START_BIT,
        DATA_BITS,
        STOP_BIT,
        ERROR
    } state_t;

    state_t state, next_state;
    reg [3:0] bit_count;
    reg [7:0] data_byte;

    always @(posedge clk or posedge reset) begin
        if (reset) begin
            state <= IDLE;
            bit_count <= 0;
            data_byte <= 0;
            done <= 0;
        end else begin
            state <= next_state;
            case (state)
                IDLE: begin
                    if (in == 0) begin
                        next_state = START_BIT;
                    end else begin
                        next_state = IDLE;
                    end
                end
                START_BIT: begin
                    if (in == 0) begin
                        next_state = DATA_BITS;
                        bit_count <= 0;
                    end else begin
                        next_state = IDLE;
                    end
                end
                DATA_BITS: begin
                    if (bit_count < 8) begin
                        data_byte <= {data_byte[6:0], in};

                        if (bit_count == 7) begin
                            next_state = STOP_BIT;
                        end else begin
                            next_state = DATA_BITS;
                        end
                    end else begin
                        next_state = ERROR;
                    end
                end
                STOP_BIT: begin
                    if (in == 1) begin
                        done <= 1;
                        next_state = IDLE;
                    end else begin
                        next_state = ERROR;
                    end
                end
                ERROR: begin
                    if (in == 1) begin
                        next_state = IDLE;
                    end else begin
                        next_state = ERROR;
                    end
                end

            endcase
