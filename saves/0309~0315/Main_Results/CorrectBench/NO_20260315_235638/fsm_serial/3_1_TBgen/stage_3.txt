def generate_expected_values(test_vector, reset_cycles=None):
    """
    Generate the expected values for the DUT based on the given test vector and reset cycles.
    
    :param test_vector: List of input bits (0s and 1s) representing the serial input stream.
    :param reset_cycles: List of clock cycles where the reset signal is high. If None, no reset is applied.
    :return: A list of expected 'done' signal values for each clock cycle.
    """
    state = 'IDLE'
    data_bits = []
    expected_done = []
    reset_active = False
    reset_counter = 0

    for i, bit in enumerate(test_vector):
        if reset_cycles and i in reset_cycles:
            reset_active = True
            reset_counter = 2  # Reset for 2 clock cycles
            state = 'IDLE'
            data_bits = []

        if reset_active:
            expected_done.append(0)
            reset_counter -= 1
            if reset_counter == 0:
                reset_active = False
            continue

        if state == 'IDLE':
            if bit == 0:
                state = 'START_BIT'
                expected_done.append(0)
            else:
                expected_done.append(0)
        elif state == 'START_BIT':

                state = 'DATA_0'
                expected_done.append(0)
            else:
                state = 'IDLE'
                expected_done.append(0)
        elif state == 'DATA_0':
