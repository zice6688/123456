import numpy as np

def generate_expected_values(input_bits, reset_signal):
    # Initialize state variables
    state = 'IDLE'
    bit_count = 0
    data_bits = []
    expected_done = []

    for i in range(len(input_bits)):
        in_bit = input_bits[i]
        reset = reset_signal[i]

        if reset:
            state = 'IDLE'
            bit_count = 0
            data_bits = []
            expected_done.append(0)
            continue

        if state == 'IDLE':
            if in_bit == 0:  # Start bit detected
                state = 'RECEIVING_DATA'
                bit_count = 0
                data_bits = []
            expected_done.append(0)
        elif state == 'RECEIVING_DATA':
            data_bits.append(in_bit)
            bit_count += 1
            if bit_count == 8:
                state = 'VERIFY_STOP_BIT'
            expected_done.append(0)
        elif state == 'VERIFY_STOP_BIT':
            if in_bit == 1:  # Correct stop bit
                state = 'IDLE'
                expected_done.append(1)
            else:  # Incorrect stop bit
                state = 'WAIT_FOR_STOP_BIT'
            expected_done.append(0)
        elif state == 'WAIT_FOR_STOP_BIT':
            if in_bit == 1:  # Correct stop bit found
                state = 'IDLE'
            expected_done.append(0)

    return expected_done
