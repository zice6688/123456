import numpy as np

def generate_expected_values(serial_bits):
    """
    Generate the expected values for the 'done' signal based on the given serial bits.
    
    :param serial_bits: A list of serial bits (0s and 1s) representing the input to the DUT.
    :return: A list of expected 'done' values corresponding to each clock cycle.
    """
    # Initialize state variables
    state = 'IDLE'
    bit_count = 0
    data_bits = []
    expected_done = []

    for bit in serial_bits:
        if state == 'IDLE':
            if bit == 0:  # Start bit detected
                state = 'RECEIVING_DATA'
                bit_count = 0
                data_bits = []
            expected_done.append(0)
        
        elif state == 'RECEIVING_DATA':
            data_bits.append(bit)
            bit_count += 1
            if bit_count == 8:  # 8 data bits received
                state = 'CHECK_STOP_BIT'
            expected_done.append(0)
        
        elif state == 'CHECK_STOP_BIT':
            if bit == 1:  # Stop bit is correct
                state = 'IDLE'
                expected_done.append(1)
            else:  # Stop bit is incorrect, wait for a stop bit
                state = 'WAIT_FOR_STOP_BIT'
            expected_done.append(0)
        
        elif state == 'WAIT_FOR_STOP_BIT':
            if bit == 1:  # Stop bit found
                state = 'IDLE'
            expected_done.append(0)
    
    return expected_done
