def generate_expected_done_signal(input_stream, reset_signal):

    """
    state = 'idle'
    bit_count = 0
    data_bits = []
    expected_done_signal = []

    for in_bit, reset in zip(input_stream, reset_signal):

            state = 'idle'
            bit_count = 0
            data_bits = []
            expected_done_signal.append(0)
            continue

            if in_bit == 0:
                state = 'start_detected'
                bit_count = 0
                data_bits = []
            expected_done_signal.append(0)
        
        elif state == 'start_detected':
            if bit_count < 8:

                bit_count += 1
            else:
                if in_bit == 1:

                    expected_done_signal.append(1)

                    expected_done_signal.append(0)
            expected_done_signal.append(0)

                state = 'idle'
                expected_done_signal.append(0)
