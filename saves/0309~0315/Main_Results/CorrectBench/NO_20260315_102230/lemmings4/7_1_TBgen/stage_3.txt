# Define the states
WALK_LEFT = 0b0001

FALLING = 0b0100
DIGGING = 0b1000

# Define the function to get the next state and outputs based on the current state and inputs
def get_next_state_and_outputs(current_state, bump_left, bump_right, ground, dig, fall_count):
    if current_state == SPLATTERED:
        return SPLATTERED, 0b0000  # All outputs are 0 in SPLATTERED state

    if ground == 0:  # Lemming is falling
        if fall_count > 20:
            return SPLATTERED, 0b0000  # Splattered after falling for more than 20 cycles
        else:
            return FALLING, 0b0001  # Falling, aaah=1, walk_left=0, walk_right=0, digging=0

    if dig == 1 and ground == 1:  # Lemming is digging
        return DIGGING, 0b0010  # Digging, digging=1, walk_left=0, walk_right=0, aaah=0

    if bump_left == 1 and current_state == WALK_LEFT:
        return WALK_RIGHT, 0b0100  # Bumped left, transition to WALK_RIGHT, walk_right=1, walk_left=0, aaah=0, digging=0

    if bump_right == 1 and current_state == WALK_RIGHT:
        return WALK_LEFT, 0b0001  # Bumped right, transition to WALK_LEFT, walk_left=1, walk_right=0, aaah=0, digging=0

    if current_state == FALLING and ground == 1:
        if fall_count > 20:
            return SPLATTERED, 0b0000  # Splattered after falling for more than 20 cycles
        else:
            if (current_state & WALK_LEFT) == WALK_LEFT:
                return WALK_LEFT, 0b0001  # Resume walking left, walk_left=1, walk_right=0, aaah=0, digging=0
            else:
                return WALK_RIGHT, 0b0100  # Resume walking right, walk_right=1, walk_left=0, aaah=0, digging=0

    if current_state == DIGGING and ground == 0:
        return FALLING, 0b0001  # Transition to FALLING, aaah=1, walk_left=0, walk_right=0, digging=0

    # Default case: stay in the current state
    if current_state == WALK_LEFT:
        return WALK_LEFT, 0b0001  # Walk left, walk_left=1, walk_right=0, aaah=0, digging=0
    elif current_state == WALK_RIGHT:
        return WALK_RIGHT, 0b0100  # Walk right, walk_right=1, walk_left=0, aaah=0, digging=0
    elif current_state == FALLING:
        return FALLING, 0b0001  # Falling, aaah=1, walk_left=0, walk_right=0, digging=0

        return DIGGING, 0b0010  # Digging, digging=1, walk_left=0, walk_right=0, aaah=0
