WALK_LEFT = 0b0001

FALLING = 0b0100
DIGGING = 0b1000

fall_counter = 0

def update_state(bump_left, bump_right, ground, dig):
    global state, fall_counter
    
    # Asynchronous reset
    if areset:
        state = WALK_LEFT
        fall_counter = 0
        return (WALK_LEFT, 0, 0, 0)
    
    # Determine the next state based on current state and inputs
    if state == WALK_LEFT:
        if bump_right:
            state = WALK_RIGHT
        elif not ground:
            state = FALLING
            fall_counter = 0
        elif dig:
            state = DIGGING
        else:
            pass  # Stay in WALK_LEFT
    elif state == WALK_RIGHT:
        if bump_left:
            state = WALK_LEFT
        elif not ground:
            state = FALLING
            fall_counter = 0
        elif dig:
            state = DIGGING
        else:
            pass  # Stay in WALK_RIGHT

    walk_right = 1 if state == WALK_RIGHT else 0
    aaah = 1 if state == FALLING or state == SPLATTERED else 0
    digging = 1 if state == DIGGING else 0
