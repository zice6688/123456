import numpy as np

def fsm_rules(in_bits, reset):
    """
    This function simulates the FSM rules for the given input bit stream and reset signal.
    It returns the expected 'done' signal for each clock cycle.
    
    Parameters:
    in_bits (list of int): The input bit stream.
    reset (list of int): The reset signal.
    
    Returns:
    list of int: The expected 'done' signal for each clock cycle.
    """
    # Initialize state variables
    state = 'IDLE'
    bit_count = 0
    done_signal = []
    data_bits = []

    for i, (bit, rst) in enumerate(zip(in_bits, reset)):
        if rst == 1:
            state = 'IDLE'
            bit_count = 0
            data_bits = []
            done_signal.append(0)
        else:
            if state == 'IDLE':
                if bit == 0:
                    state = 'START'
                    bit_count = 0
                    data_bits = []
                done_signal.append(0)
            elif state == 'START':
                if bit == 0:
                    state = 'START'
                else:
                    state = 'DATA'
                    data_bits.append(bit)
                    bit_count += 1
                done_signal.append(0)
            elif state == 'DATA':
                if bit_count < 8:
                    data_bits.append(bit)
                    bit_count += 1
                else:
                    if bit == 1:
                        state = 'STOP'
                    else:
                        state = 'ERROR'
                done_signal.append(0)
            elif state == 'STOP':
                if bit == 1:
                    state = 'IDLE'
                    done_signal.append(1)
                else:
                    state = 'ERROR'
                    done_signal.append(0)
            elif state == 'ERROR':
                if bit == 1:
                    state = 'IDLE'
                done_signal.append(0)

    return done_signal

# Example usage

expected_done = fsm_rules(in_bits, reset)
print(expected_done)
