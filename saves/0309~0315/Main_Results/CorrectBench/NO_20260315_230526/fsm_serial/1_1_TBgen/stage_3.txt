import numpy as np

def generate_expected_values(clk, in_stream, reset):
    """
    Generate the expected values for the DUT based on the input signals.
    
    Parameters:
    - clk: Clock signal (list of 0s and 1s)
    - in_stream: Serial input bit stream (list of 0s and 1s)
    - reset: Active-high synchronous reset signal (list of 0s and 1s)
    
    Returns:
    - done: Expected done signal (list of 0s and 1s)
    """
    state = 'Idle'
    bit_count = 0
    data_bits = []
    done = [0] * len(clk)
    
    for i in range(len(clk)):
        if reset[i] == 1:
            state = 'Idle'
            bit_count = 0
            data_bits = []
            done[i] = 0
            continue
        
        if state == 'Idle':
            if in_stream[i] == 0:
                state = 'Start Bit Detected'
                bit_count = 0
                data_bits = []
        
        elif state == 'Start Bit Detected':
            if in_stream[i] == 0:
                state = 'Data Bits Reception'
                bit_count = 0
                data_bits = []
            else:
                state = 'Idle'
        
        elif state == 'Data Bits Reception':
            data_bits.append(in_stream[i])
            bit_count += 1
            if bit_count == 8:
                state = 'Stop Bit Verification'
        
        elif state == 'Stop Bit Verification':
            if in_stream[i] == 1:
                done[i] = 1
                state = 'Idle'
            else:
                state = 'Wait for Stop Bit'
        
        elif state == 'Wait for Stop Bit':
            if in_stream[i] == 1:
                state = 'Idle'
            else:
                state = 'Wait for Stop Bit'
    
    return done
