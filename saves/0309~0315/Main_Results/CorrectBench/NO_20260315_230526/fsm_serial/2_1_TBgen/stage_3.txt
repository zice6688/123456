import numpy as np

def fsm_rules(in_bits, reset):
    # Initialize state and counters
    state = 'IDLE'
    bit_count = 0
    done_signal = [0] * len(in_bits)
    
    for i, bit in enumerate(in_bits):
        if reset[i]:
            state = 'IDLE'
            bit_count = 0
            continue
        
        if state == 'IDLE':
            if bit == 0:
                state = 'START_BIT'
        
        elif state == 'START_BIT':
            if bit == 0:
                state = 'DATA_BITS'
                bit_count = 0
            else:
                state = 'IDLE'
        
        elif state == 'DATA_BITS':
            bit_count += 1
            if bit_count == 8:
                state = 'STOP_BIT'
        
        elif state == 'STOP_BIT':
            if bit == 1:
                state = 'IDLE'
                done_signal[i] = 1
            else:
                state = 'ERROR'
        
        elif state == 'ERROR':
            if bit == 1:
                state = 'IDLE'
    
    return done_signal
