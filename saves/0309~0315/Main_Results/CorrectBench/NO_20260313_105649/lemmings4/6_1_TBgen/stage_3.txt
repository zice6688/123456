import numpy as np

def walking_left(bump_left, bump_right, ground, dig):
    if bump_left:
        return 0b0010  # Walking right
    elif ground == 0:
        return 0b0100  # Falling
    elif dig and ground == 1:
        return 0b1000  # Digging
    else:
        return 0b0001  # Stay in walking left

def walking_right(bump_left, bump_right, ground, dig):
    if bump_right:
        return 0b0001  # Walking left
    elif ground == 0:
        return 0b0100  # Falling
    elif dig and ground == 1:
        return 0b1000  # Digging
    else:
        return 0b0010  # Stay in walking right

def falling(ground, fall_count):
    if ground == 1:
        if fall_count > 20:
            return 0b0000  # Splattered
        else:
            return 0b0001  # Resume walking left
    else:
        return 0b0100  # Continue falling

def digging(ground):
    if ground == 0:
        return 0b0100  # Falling
    else:
        return 0b1000  # Stay in digging

def get_expected_values(current_state, bump_left, bump_right, ground, dig, fall_count):
    if current_state == 0b0001:  # Walking left
        return walking_left(bump_left, bump_right, ground, dig)
    elif current_state == 0b0010:  # Walking right
        return walking_right(bump_left, bump_right, ground, dig)
    elif current_state == 0b0100:  # Falling
        return falling(ground, fall_count)
    elif current_state == 0b1000:  # Digging
        return digging(ground)
    elif current_state == 0b0000:  # Splattered
        return 0b0000  # Stay in splattered state
    else:
        raise ValueError("Invalid state")

# Example usage
current_state = 0b0001  # Start in walking left
bump_left = 0
bump_right = 0
ground = 1
dig = 0
fall_count = 0

next_state = get_expected_values(current_state, bump_left, bump_right, ground, dig, fall_count)
