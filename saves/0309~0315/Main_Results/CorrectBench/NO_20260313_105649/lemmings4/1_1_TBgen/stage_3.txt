import numpy as np

def lemming_rules(bump_left, bump_right, ground, dig, current_state, fall_counter):
    """
    This function implements the core rules for the Lemming's behavior.
    
    Parameters:
    - bump_left (int): 1 if bumped on the left, 0 otherwise.
    - bump_right (int): 1 if bumped on the right, 0 otherwise.
    - ground (int): 1 if ground is present, 0 if falling.
    - dig (int): 1 if digging, 0 otherwise.
    - current_state (tuple): A tuple (walk_left, walk_right, aaah, digging) representing the current state of the Lemming.
    - fall_counter (int): A counter to track the number of clock cycles the Lemming has been falling.
    
    Returns:
    - new_state (tuple): A tuple (walk_left, walk_right, aaah, digging) representing the new state of the Lemming.
    - new_fall_counter (int): The updated fall counter.
    """
    walk_left, walk_right, aaah, digging = current_state
    
    # Initialize new state and fall counter
    new_walk_left = walk_left
    new_walk_right = walk_right
    new_aaah = aaah
    new_digging = digging
    new_fall_counter = fall_counter
    
    # Check for splattering condition
    if fall_counter > 20 and ground == 1:
        return (0, 0, 0, 0), 0
    
    # Handle falling
    if ground == 0:
        new_aaah = 1
        new_walk_left = 0
        new_walk_right = 0
        new_digging = 0
        new_fall_counter += 1
    else:
        new_aaah = 0
        new_fall_counter = 0
        
        # Handle digging
        if dig == 1 and ground == 1 and not aaah:
            new_digging = 1
            new_walk_left = 0
            new_walk_right = 0
        else:
            new_digging = 0
        
        # Handle walking and direction changes
        if bump_left == 1 or bump_right == 1:
            if bump_left == 1:
                new_walk_left = 0
                new_walk_right = 1
            elif bump_right == 1:
                new_walk_left = 1
                new_walk_right = 0
        else:
            if walk_left == 1:
                new_walk_left = 1
                new_walk_right = 0
            elif walk_right == 1:
                new_walk_left = 0
                new_walk_right = 1
    
    return (new_walk_left, new_walk_right, new_aaah, new_digging), new_fall_counter
