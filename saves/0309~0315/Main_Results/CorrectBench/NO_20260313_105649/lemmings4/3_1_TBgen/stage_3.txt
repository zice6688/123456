def lemming_rules(bump_left, bump_right, ground, dig, current_state, fall_counter):
    """
    This function defines the core rules for the Lemming's behavior based on the given inputs.
    
    Parameters:
    - bump_left: 1 if the Lemming is bumped from the left, 0 otherwise.
    - bump_right: 1 if the Lemming is bumped from the right, 0 otherwise.
    - ground: 1 if there is ground, 0 otherwise.
    - dig: 1 if the Lemming is commanded to dig, 0 otherwise.
    - current_state: A dictionary containing the current state of the Lemming.
      - 'walk_left': 1 if the Lemming is walking left, 0 otherwise.
      - 'walk_right': 1 if the Lemming is walking right, 0 otherwise.
      - 'aaah': 1 if the Lemming is falling and saying 'aaah!', 0 otherwise.
      - 'digging': 1 if the Lemming is digging, 0 otherwise.
    - fall_counter: The number of clock cycles the Lemming has been falling.
    
    Returns:
    - A dictionary containing the expected values for the outputs.
      - 'walk_left': 1 if the Lemming should be walking left, 0 otherwise.
      - 'walk_right': 1 if the Lemming should be walking right, 0 otherwise.
      - 'aaah': 1 if the Lemming should be falling and saying 'aaah!', 0 otherwise.
      - 'digging': 1 if the Lemming should be digging, 0 otherwise.
    """
    # Initialize the expected output values
    expected = {
        'walk_left': 0,
        'walk_right': 0,
        'aaah': 0,
        'digging': 0
    }
    
    # Check if the Lemming is splattered
    if fall_counter > 20 and ground == 1:
        return expected  # All outputs are 0 if the Lemming splatters
    
    # Check if the Lemming is falling
    if ground == 0:
        expected['aaah'] = 1
        return expected  # If falling, only 'aaah' is 1, others are 0
    
    # Check if the Lemming is digging
    if dig == 1 and ground == 1 and not (current_state['aaah'] or current_state['digging']):
        expected['digging'] = 1
        return expected  # If digging, only 'digging' is 1, others are 0
    
    # Check if the Lemming is bumped
    if bump_left == 1 or bump_right == 1:
        if bump_left == 1 and bump_right == 0:
            expected['walk_right'] = 1
        elif bump_left == 0 and bump_right == 1:
            expected['walk_left'] = 1
        else:  # Both bump_left and bump_right are 1
            if current_state['walk_left']:
                expected['walk_right'] = 1
            else:
                expected['walk_left'] = 1
        return expected  # If bumped, set the appropriate walk direction
    
    # If no special conditions, maintain the current walking direction
    if current_state['walk_left']:
        expected['walk_left'] = 1
    elif current_state['walk_right']:
        expected['walk_right'] = 1
    
    return expected
