import numpy as np

def lemming_rules(bump_left, bump_right, ground, dig, current_state, fall_counter):
    """
    This function implements the core rules for the Lemmings state machine.
    
    Parameters:
    - bump_left: 1 if bumped on the left, 0 otherwise
    - bump_right: 1 if bumped on the right, 0 otherwise
    - ground: 1 if there is ground, 0 otherwise
    - dig: 1 if the Lemming should start digging, 0 otherwise
    - current_state: A tuple (walk_left, walk_right, aaah, digging) representing the current state
    - fall_counter: The number of clock cycles the Lemming has been falling
    
    Returns:
    - new_state: A tuple (walk_left, walk_right, aaah, digging) representing the new state
    - new_fall_counter: The updated fall counter
    """
    walk_left, walk_right, aaah, digging = current_state
    
    # Determine the new state based on the inputs and current state
    if not ground:
        # If there is no ground, the Lemming starts falling
        if not aaah:
            # Start falling
            new_state = (0, 0, 1, 0)
            new_fall_counter = 1
        else:
            # Continue falling
            new_state = (0, 0, 1, 0)
            new_fall_counter = fall_counter + 1
            if new_fall_counter > 20 and ground:
                # Splatter if falling for more than 20 cycles and hits the ground
                new_state = (0, 0, 0, 0)
                new_fall_counter = 0
    elif ground and aaah:
        # If the ground reappears after falling, resume the previous walking direction
        if walk_left == 1:
            new_state = (1, 0, 0, 0)
        else:
            new_state = (0, 1, 0, 0)
        new_fall_counter = 0
    elif dig and not aaah and not digging:
        # If the Lemming is on the ground and not falling or digging, start digging
        new_state = (0, 0, 0, 1)
        new_fall_counter = 0
    elif digging and not ground:
        # If the Lemming is digging and the ground disappears, start falling
        new_state = (0, 0, 1, 0)
        new_fall_counter = 1
    elif bump_left and not aaah and not digging:
        # If the Lemming is bumped on the left, switch to walking right
        new_state = (0, 1, 0, 0)
        new_fall_counter = 0
    elif bump_right and not aaah and not digging:
        # If the Lemming is bumped on the right, switch to walking left
        new_state = (1, 0, 0, 0)
        new_fall_counter = 0
    else:
        # No change in state
        new_state = (walk_left, walk_right, aaah, digging)
        new_fall_counter = fall_counter
    
    return new_state, new_fall_counter
