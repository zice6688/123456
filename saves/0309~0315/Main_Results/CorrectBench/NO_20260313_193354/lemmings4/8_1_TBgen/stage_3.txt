import numpy as np

def lemming_fsm(bump_left, bump_right, ground, dig, current_state, fall_counter):
    # Initialize the outputs
    walk_left = 0
    walk_right = 0
    aaah = 0
    digging = 0
    
    # Current state decoding
    if current_state == 0b0001:  # Walking left
        if bump_left and not (dig or (not ground)):
            walk_left = 0
            walk_right = 1
            current_state = 0b0010  # Switch to walking right
        elif bump_right and not (dig or (not ground)):
            walk_left = 1
            walk_right = 0
            current_state = 0b0001  # Stay in walking left
        elif dig and ground:
            walk_left = 0
            walk_right = 0
            digging = 1
            current_state = 0b0100  # Start digging
        elif not ground:
            walk_left = 0
            walk_right = 0
            aaah = 1
            current_state = 0b1000  # Start falling
        else:
            walk_left = 1
            walk_right = 0
            current_state = 0b0001  # Stay in walking left
    elif current_state == 0b0010:  # Walking right
        if bump_left and not (dig or (not ground)):
            walk_left = 0
            walk_right = 1
            current_state = 0b0010  # Stay in walking right
        elif bump_right and not (dig or (not ground)):
            walk_left = 1
            walk_right = 0
            current_state = 0b0001  # Switch to walking left
        elif dig and ground:
            walk_left = 0
            walk_right = 0
            digging = 1
            current_state = 0b0100  # Start digging
        elif not ground:
            walk_left = 0
            walk_right = 0
            aaah = 1
            current_state = 0b1000  # Start falling
        else:
            walk_left = 0
            walk_right = 1
            current_state = 0b0010  # Stay in walking right
    elif current_state == 0b0100:  # Digging
        if not ground:
            walk_left = 0
            walk_right = 0
            digging = 0
            aaah = 1
            current_state = 0b1000  # Start falling
        else:
            walk_left = 0
            walk_right = 0
            digging = 1
            current_state = 0b0100  # Continue digging
    elif current_state == 0b1000:  # Falling
        if ground and fall_counter > 20:
            walk_left = 0
            walk_right = 0
            aaah = 0
            digging = 0
            current_state = 0b0000  # Splatter
        elif ground:
            if current_state == 0b0001:
                walk_left = 1
                walk_right = 0
                current_state = 0b0001  # Resume walking left
            elif current_state == 0b0010:
                walk_left = 0
                walk_right = 1
                current_state = 0b0010  # Resume walking right
            aaah = 0
            digging = 0
        else:
            walk_left = 0
            walk_right = 0
            aaah = 1
            digging = 0
            current_state = 0b1000  # Continue falling
            fall_counter += 1
    elif current_state == 0b0000:  # Splattered
        walk_left = 0
        walk_right = 0
        aaah = 0
        digging = 0
        current_state = 0b0000  # Stay splattered

    return walk_left, walk_right, aaah, digging, current_state, fall_counter
