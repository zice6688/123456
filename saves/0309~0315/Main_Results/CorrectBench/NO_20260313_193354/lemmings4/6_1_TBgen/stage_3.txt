import numpy as np

def lemming_rules(current_state, bump_left, bump_right, ground, dig, fall_counter):
    """
    This function defines the core rules for the Lemming's behavior based on the given inputs.
    
    Parameters:
    - current_state: A 4-bit binary string representing the current state of the Lemming.
                     The bits represent [walk_left, walk_right, aaah, digging] respectively.
    - bump_left: A boolean indicating if the Lemming is bumped from the left.
    - bump_right: A boolean indicating if the Lemming is bumped from the right.
    - ground: A boolean indicating if the Lemming is on the ground.
    - dig: A boolean indicating if the Lemming should start digging.
    - fall_counter: An integer counter to track the number of clock cycles the Lemming has been falling.
    
    Returns:
    - next_state: A 4-bit binary string representing the next state of the Lemming.
    - new_fall_counter: The updated fall counter.
    - splattered: A boolean indicating if the Lemming has splattered.
    """
    
    # Unpack the current state
    walk_left, walk_right, aaah, digging = map(int, current_state)
    
    # Initialize the next state and new fall counter
    next_state = current_state
    new_fall_counter = fall_counter
    splattered = False
    
    # Handle asynchronous reset
    if areset:
        return "0b1000", 0, False  # Reset to walking left
    
    # Check for falling
    if not ground:
        if not aaah:
            new_fall_counter += 1
        next_state = "0b0010"  # Falling (aaah = 1, walk_left = 0, walk_right = 0, digging = 0)
        if new_fall_counter > 20 and ground:
            splattered = True
            next_state = "0b0000"  # Splattered (all outputs = 0)
    
    # Check for resuming walking after falling
    elif ground and aaah:
        if walk_left:
            next_state = "0b1000"  # Resume walking left
        else:
            next_state = "0b0100"  # Resume walking right
        new_fall_counter = 0
    
    # Check for digging
    if ground and dig and not aaah and not digging:
        next_state = "0b0001"  # Start digging (digging = 1, walk_left = 0, walk_right = 0, aaah = 0)
    
    # Check for switching directions
    if bump_left and walk_left and not aaah and not digging:
        next_state = "0b0100"  # Switch to walking right
    elif bump_right and walk_right and not aaah and not digging:
        next_state = "0b1000"  # Switch to walking left
    elif bump_left and bump_right and not aaah and not digging:
        if walk_left:
            next_state = "0b0100"  # Switch to walking right
        else:
            next_state = "0b1000"  # Switch to walking left
    
    return next_state, new_fall_counter, splattered
