import numpy as np

def lemming_rules(current_state, bump_left, bump_right, ground, dig, fall_counter):
    """
    This function implements the core rules for the Lemming's behavior.
    
    :param current_state: A 4-bit binary string representing the current state of the Lemming:
                          [walk_left, walk_right, aaah, digging]
    :param bump_left:     1 if the Lemming is bumped on the left, 0 otherwise
    :param bump_right:    1 if the Lemming is bumped on the right, 0 otherwise
    :param ground:        1 if the Lemming is on the ground, 0 if it is falling
    :param dig:           1 if the Lemming is commanded to dig, 0 otherwise
    :param fall_counter:  The number of clock cycles the Lemming has been falling
    :return: A 4-bit binary string representing the next state of the Lemming:
             [walk_left, walk_right, aaah, digging]
    """
    # Convert the current state from a binary string to a list of integers
    walk_left, walk_right, aaah, digging = [int(bit) for bit in current_state]

    # Determine the next state based on the inputs and the current state
    if ground == 0:
        # If the ground disappears, the Lemming falls
        if not aaah:
            aaah = 1
            walk_left = 0
            walk_right = 0
            digging = 0
            fall_counter = 1
        else:
            fall_counter += 1
            if fall_counter > 20 and ground == 1:
                # If the Lemming falls for more than 20 cycles and then hits the ground, it splatters
                return "0b0000"
    else:
        # If the ground reappears, the Lemming resumes walking or digging
        if aaah:
            aaah = 0
            if walk_left:
                walk_left = 1
                walk_right = 0
            elif walk_right:
                walk_right = 1
                walk_left = 0
            elif digging:
                digging = 1
        elif dig and not (aaah or digging):
            # If the Lemming is on the ground and receives a dig command, it starts digging
            digging = 1
            walk_left = 0
            walk_right = 0
        elif bump_left and not (aaah or digging):
            # If the Lemming is bumped on the left, it switches to walking right
            walk_left = 0
            walk_right = 1
        elif bump_right and not (aaah or digging):
            # If the Lemming is bumped on the right, it switches to walking left
            walk_left = 1
            walk_right = 0
        elif bump_left and bump_right and not (aaah or digging):
            # If the Lemming is bumped on both sides, it switches direction
            if walk_left:
                walk_left = 0
                walk_right = 1
            else:
                walk_left = 1
                walk_right = 0

    # Convert the next state back to a 4-bit binary string
    next_state = f"0b{walk_left}{walk_right}{aaah}{digging}"
    return next_state
