import numpy as np

def lemming_fsm(bump_left, bump_right, ground, dig, current_state, fall_counter):
    """
    This function implements the core rules of the Lemming's FSM.
    
    Parameters:
    - bump_left: 1 if the Lemming is bumped on the left, 0 otherwise.
    - bump_right: 1 if the Lemming is bumped on the right, 0 otherwise.
    - ground: 1 if the Lemming is on the ground, 0 if it is falling.
    - dig: 1 if the Lemming is instructed to dig, 0 otherwise.
    - current_state: A tuple (walk_left, walk_right, aaah, digging) representing the current state.
    - fall_counter: The number of clock cycles the Lemming has been falling.
    
    Returns:
    - new_state: A tuple (walk_left, walk_right, aaah, digging) representing the new state.
    - new_fall_counter: The updated fall counter.
    - splatter: 1 if the Lemming splatters, 0 otherwise.
    """
    walk_left, walk_right, aaah, digging = current_state
    
    # Initialize new state and splatter flag
    new_walk_left = walk_left
    new_walk_right = walk_right
    new_aaah = aaah
    new_digging = digging
    splatter = 0
    
    # Update fall counter
    if not ground:
        new_fall_counter = fall_counter + 1
    else:
        new_fall_counter = 0
    
    # Check for splatter
    if new_fall_counter > 20 and ground:
        splatter = 1
        new_walk_left = 0
        new_walk_right = 0
        new_aaah = 0
        new_digging = 0
        return (new_walk_left, new_walk_right, new_aaah, new_digging), new_fall_counter, splatter
    
    # Handle bumping
    if bump_left or bump_right:
        if bump_left:
            new_walk_left = 0
            new_walk_right = 1
        elif bump_right:
            new_walk_left = 1
            new_walk_right = 0
    
    # Handle falling
    if not ground:
        new_walk_left = 0
        new_walk_right = 0
        new_aaah = 1
        new_digging = 0
    else:
        new_aaah = 0
        if dig and not new_digging:
            new_digging = 1
            new_walk_left = 0
            new_walk_right = 0
        elif not dig and new_digging:
            new_digging = 0
            new_walk_left = walk_left
            new_walk_right = walk_right
    
    # Return the new state, fall counter, and splatter flag
    return (new_walk_left, new_walk_right, new_aaah, new_digging), new_fall_counter, splatter
