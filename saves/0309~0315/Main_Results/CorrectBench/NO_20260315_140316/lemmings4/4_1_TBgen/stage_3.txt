import numpy as np

def lemming_rules(bump_left, bump_right, ground, dig, current_state, fall_counter):
    """
    This function defines the core rules for the Lemming's finite state machine.
    
    Parameters:
    - bump_left: 1 if bumped on the left, 0 otherwise
    - bump_right: 1 if bumped on the right, 0 otherwise
    - ground: 1 if there is ground, 0 otherwise
    - dig: 1 if the Lemming should start digging, 0 otherwise
    - current_state: The current state of the Lemming (WL, WR, F, D, S)
    - fall_counter: Counter to track the number of clock cycles the Lemming has been falling
    
    Returns:
    - next_state: The next state of the Lemming
    - walk_left: 1 if walking left, 0 otherwise
    - walk_right: 1 if walking right, 0 otherwise
    - aaah: 1 if falling, 0 otherwise
    - digging: 1 if digging, 0 otherwise
    - fall_counter: Updated fall counter
    """
    
    # Define states
    WL = 0b0001  # Walking Left
    WR = 0b0010  # Walking Right
    F = 0b0100   # Falling
    D = 0b1000   # Digging
    S = 0b10000  # Splattered

    # Initialize outputs
    walk_left = 0
    walk_right = 0
    aaah = 0
    digging = 0

    # Determine the next state and outputs based on the current state and inputs
    if current_state == WL:
        if bump_left:
            next_state = WR
        elif not ground:
            next_state = F
            fall_counter = 0
        elif dig and ground:
            next_state = D
        else:
            next_state = WL
            walk_left = 1
    elif current_state == WR:
        if bump_right:
            next_state = WL
        elif not ground:
            next_state = F
            fall_counter = 0
        elif dig and ground:
            next_state = D
        else:
            next_state = WR
            walk_right = 1
    elif current_state == F:
        if ground:
            if fall_counter >= 20:
                next_state = S
            else:
                next_state = current_state & ~F  # Resume previous direction
        else:
            fall_counter += 1
            next_state = F
        aaah = 1
    elif current_state == D:
        if not ground:
            next_state = F
            fall_counter = 0
        else:
            next_state = D
            digging = 1
    elif current_state == S:
        next_state = S
    else:
        raise ValueError("Invalid current state")

    return next_state, walk_left, walk_right, aaah, digging, fall_counter
