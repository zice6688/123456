import numpy as np

def lemming_rules(current_state, bump_left, bump_right, ground, dig, fall_counter):
    """
    This function defines the core rules for the Lemming's behavior based on the given inputs.
    
    Parameters:
    - current_state: A dictionary representing the current state of the Lemming.
        - 'walk_left': 1 if walking left, 0 otherwise.
        - 'walk_right': 1 if walking right, 0 otherwise.
        - 'aaah': 1 if falling, 0 otherwise.
        - 'digging': 1 if digging, 0 otherwise.
        - 'splattered': 1 if splattered, 0 otherwise.
    - bump_left: 1 if bumped on the left, 0 otherwise.
    - bump_right: 1 if bumped on the right, 0 otherwise.
    - ground: 1 if there is ground, 0 otherwise.
    - dig: 1 if the Lemming is told to dig, 0 otherwise.
    - fall_counter: The number of clock cycles the Lemming has been falling.
    
    Returns:
    - new_state: A dictionary representing the new state of the Lemming after applying the rules.
    - new_fall_counter: The updated fall counter.
    """
    
    # Initialize the new state with the current state
    new_state = current_state.copy()
    new_fall_counter = fall_counter
    
    # Check if the Lemming is splattered
    if current_state['splattered']:
        return new_state, new_fall_counter
    
    # Check for falling
    if not ground:
        new_state['aaah'] = 1
        new_state['walk_left'] = 0
        new_state['walk_right'] = 0
        new_state['digging'] = 0
        new_fall_counter += 1
    else:
        new_fall_counter = 0
    
    # Check for splattering
    if new_fall_counter > 20 and ground:
        new_state['walk_left'] = 0
        new_state['walk_right'] = 0
        new_state['aaah'] = 0
        new_state['digging'] = 0
        new_state['splattered'] = 1
        return new_state, new_fall_counter
    
    # Check for digging
    if dig and ground and not new_state['aaah']:
        new_state['digging'] = 1
        new_state['walk_left'] = 0
        new_state['walk_right'] = 0
        new_state['aaah'] = 0
    elif not ground and new_state['digging']:
        new_state['digging'] = 0
        new_state['aaah'] = 1
        new_state['walk_left'] = 0
        new_state['walk_right'] = 0
    
    # Check for walking direction changes
    if not new_state['aaah'] and not new_state['digging']:
        if bump_left:
            new_state['walk_left'] = 0
            new_state['walk_right'] = 1
        elif bump_right:
            new_state['walk_left'] = 1
            new_state['walk_right'] = 0
    
    # Resume previous walking direction when ground reappears
    if ground and new_state['aaah']:
        if current_state['walk_left']:
            new_state['walk_left'] = 1
            new_state['walk_right'] = 0
        elif current_state['walk_right']:
            new_state['walk_left'] = 0
            new_state['walk_right'] = 1
        new_state['aaah'] = 0
    
    return new_state, new_fall_counter
