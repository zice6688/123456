WL = 0b0001  # Walking Left
WR = 0b0010  # Walking Right
F = 0b0100   # Falling
D = 0b1000   # Digging
S = 0b10000  # Splattered

def lemming_fsm(current_state, bump_left, bump_right, ground, dig, fall_count):
    next_state = current_state
    walk_left = 0
    walk_right = 0
    aaah = 0
    digging = 0

    if current_state == S:
        return S, 0, 0, 0, 0

    if bump_left == 1 and bump_right == 1:
        if current_state == WL:
            next_state = WR
        elif current_state == WR:
            next_state = WL
    elif bump_left == 1:
        next_state = WR
    elif bump_right == 1:
        next_state = WL

    if ground == 0:
        if current_state != F:
            next_state = F
            fall_count = 0
        else:
            fall_count += 1
        if fall_count > 20 and ground == 1:
            next_state = S
    elif ground == 1:
        if current_state == F:
            if fall_count > 20:
                next_state = S
            else:
                next_state = current_state
        elif dig == 1 and (current_state == WL or current_state == WR):
            next_state = D

            next_state = current_state

        walk_left = 1
    elif next_state == WR:
        walk_right = 1
    elif next_state == F:
        aaah = 1
    elif next_state == D:
        digging = 1

    return next_state, walk_left, walk_right, aaah, digging
