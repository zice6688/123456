import numpy as np

def lemming_rules(bump_left, bump_right, ground, dig, current_state, fall_counter):
    """
    This function implements the core rules for the Lemming's behavior based on the given inputs.
    
    Parameters:
    - bump_left: 1 if the Lemming is bumped on the left, 0 otherwise.
    - bump_right: 1 if the Lemming is bumped on the right, 0 otherwise.
    - ground: 1 if there is ground, 0 otherwise.
    - dig: 1 if the Lemming is told to dig, 0 otherwise.
    - current_state: A tuple (walk_left, walk_right, aaah, digging) representing the current state of the Lemming.
    - fall_counter: The number of clock cycles the Lemming has been falling.
    
    Returns:
    - next_state: A tuple (walk_left, walk_right, aaah, digging) representing the next state of the Lemming.
    - new_fall_counter: The updated fall counter.
    """
    walk_left, walk_right, aaah, digging = current_state
    
    # Determine the next state based on the current state and inputs
    if areset:
        # Asynchronous reset: set the Lemming to walking left
        next_state = (1, 0, 0, 0)
        new_fall_counter = 0
    elif not ground:
        # Lemming is falling
        if fall_counter >= 20:
            # Lemming splatters after 20 clock cycles of falling
            next_state = (0, 0, 0, 0)
            new_fall_counter = 0
        else:
            # Lemming continues falling
            next_state = (0, 0, 1, 0)
            new_fall_counter = fall_counter + 1
    elif dig and ground and not aaah and not digging:
        # Lemming starts digging
        next_state = (0, 0, 0, 1)
        new_fall_counter = 0
    elif digging and not ground:
        # Lemming falls after digging
        next_state = (0, 0, 1, 0)
        new_fall_counter = 1
    elif bump_left and walk_left and not aaah and not digging:
        # Lemming switches to walking right when bumped on the left
        next_state = (0, 1, 0, 0)
        new_fall_counter = 0
    elif bump_right and walk_right and not aaah and not digging:
        # Lemming switches to walking left when bumped on the right
        next_state = (1, 0, 0, 0)
        new_fall_counter = 0
    else:
        # Lemming continues in its current state
        next_state = (walk_left, walk_right, aaah, digging)
        new_fall_counter = 0 if ground else fall_counter + 1
    
    return next_state, new_fall_counter
